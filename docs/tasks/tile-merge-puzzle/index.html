Tile Merge Puzzle (2048-style)

Build a complete playable 2048-style tile-merging game in a single HTML file. Compare game logic, UI polish, and whether the result is actually playable.

Prompt v1.0.0

Build a **complete, playable** tile-merging puzzle game in the style of the famous **2048** puzzle (MIT-licensed original — use only generic mechanics, no copied assets or branding).

## Game concept

A 4×4 grid where numbered tiles slide and merge. Combine matching tiles to reach higher values. The game is famous worldwide — implement the core loop faithfully while using your own visual design.

## Required game parts (all must work)

1. **Title screen** — game name, short instructions, "Press Space or click to start"
2. **Play state** — 4×4 grid, score display, best score (persist with `localStorage`)
3. **Tile logic** — spawn a `2` (90%) or `4` (10%) after each valid move; merge equal adjacent tiles in the moved direction (one merge per tile per move)
4. **Controls** — arrow keys (and optional swipe on touch)
5. **Win detection** — reaching the 2048 tile shows a win overlay with "Keep playing" and "New game"
6. **Game over** — no empty cells and no possible merges; show game-over overlay with final score and "Try again"
7. **New game** — reset grid, score, and state from title or overlays

## Technical requirements

1. Output **only** a single fenced `html` code block containing the full page. No explanation outside the code block.
2. Self-contained: embedded CSS and JavaScript only. No external libraries, fonts, or images.
3. Use semantic HTML5. The game must be **immediately playable** when the page loads in a browser.
4. Smooth but lightweight UI — colored tiles by value (e.g. 2=light, 4=warm, 8=orange, higher=darker), readable numbers, subtle transitions.
5. Include a visible **controls hint** on screen (e.g. "Arrow keys to move").
6. Do **not** implement Snake, Flappy Bird, Tetris, Pac-Man, or other unrelated games.
7. Do **not** use copyrighted sprites, sound files, or trademarked names beyond a generic title like "Tile Merge" or "2048 Puzzle".

The result should feel like a polished indie browser game — not a broken prototype.

(function(){
  "use strict";
  const SIZE = 4;
  const WIN_VALUE = 2048;
  const STORAGE_BEST = "tilemerge.best";

  const boardEl   = document.getElementById("board");
  const tilesEl   = document.getElementById("tiles");
  const gridBgEl  = document.getElementById("gridBg");
  const scoreEl   = document.getElementById("score");
  const bestEl    = document.getElementById("best");
  const finalEl   = document.getElementById("finalScore");

  const titleOverlay = document.getElementById("titleOverlay");
  const winOverlay   = document.getElementById("winOverlay");
  const overOverlay  = document.getElementById("overOverlay");

  let grid = [];          // 4x4 of tile objects or null
  let score = 0;
  let best = 0;
  let idSeq = 1;
  let state = "title";    // title | playing | won | over
  let keepPlaying = false;

  // build background cells
  for (let i=0;i<SIZE*SIZE;i++){
    const c = document.createElement("div");
    c.className = "cell";
    gridBgEl.appendChild(c);
  }

  best = parseInt(localStorage.getItem(STORAGE_BEST) || "0", 10) || 0;
  bestEl.textContent = best;

  function emptyGrid(){
    grid = [];
    for (let r=0;r<SIZE;r++){ grid.push(new Array(SIZE).fill(null)); }
  }

  function randomEmptyCell(){
    const cells = [];
    for (let r=0;r<SIZE;r++)
      for (let c=0;c<SIZE;c++)
        if (!grid[r][c]) cells.push({r,c});
    if (!cells.length) return null;
    return cells[Math.floor(Math.random()*cells.length)];
  }

  function spawnTile(){
    const cell = randomEmptyCell();
    if (!cell) return null;
    const value = Math.random() < 0.9 ? 2 : 4;
    const tile = { id: idSeq++, value, r: cell.r, c: cell.c, isNew:true, merged:false };
    grid[cell.r][cell.c] = tile;
    return tile;
  }

  // ---- rendering ----
  function tilePercent(index){
    // each tile occupies (100 - gaps)/4 ; compute translate offset in %
    return index * (100/SIZE);
  }

  function valClass(v){
    if (v >= 4096) return "vbig";
    return "v"+v;
  }

  function render(){
    tilesEl.innerHTML = "";
    const cellPct = 100/SIZE;           // fraction of tiles-container per cell
    const gapPx = parseInt(getComputedStyle(document.documentElement).getPropertyValue('--gap')) || 12;
    for (let r=0;r<SIZE;r++){
      for (let c=0;c<SIZE;c++){
        const t = grid[r][c];
        if (!t) continue;
        const el = document.createElement("div");
        el.className = "tile " + valClass(t.value);
        // size: each cell minus gap spacing handled via calc
        el.style.width  = `calc((100% - ${(SIZE-1)*gapPx}px) / ${SIZE})`;
        el.style.height = el.style.width;
        const tx = `calc((100% + ${gapPx}px) * ${c})`;
        const ty = `calc((100% + ${gapPx}px) * ${r})`;
        el.style.setProperty("--tx", `calc(${c} * (100% + ${gapPx}px))`);
        el.style.setProperty("--ty", `calc(${r} * (100% + ${gapPx}px))`);
        // position via transform relative to its own size
        el.style.transform = `translate(calc(${c} * (100% + ${gapPx}px)), calc(${r} * (100% + ${gapPx}px)))`;
        const digits = String(t.value).length;
        el.style.fontSize = digits >= 4 ? "1.35rem" : digits === 3 ? "1.7rem" : "2.1rem";
        const n = document.createElement("span");
        n.className = "n";
        n.textContent = t.value;
        el.appendChild(n);
        if (t.isNew) el.classList.add("spawn");
        if (t.merged) el.classList.add("merged");
        tilesEl.appendChild(el);
      }
    }
    scoreEl.textContent = score;
    if (score > best){
      best = score;
      bestEl.textContent = best;
      localStorage.setItem(STORAGE_BEST, String(best));
    }
  }

  function clearFlags(){
    for (let r=0;r<SIZE;r++)
      for (let c=0;c<SIZE;c++)
        if (grid[r][c]){ grid[r][c].isNew=false; grid[r][c].merged=false; }
  }

  // ---- movement ----
  // Build traversal order and slide. dir: 'up','down','left','right'
  function move(dir){
    if (state !== "playing" && state !== "won") return;
    clearFlags();

    const vector = {left:[0,-1],right:[0,1],up:[-1,0],down:[1,0]}[dir];
    const rows = [...Array(SIZE).keys()];
    const cols = [...Array(SIZE).keys()];
    if (vector[0] === 1) rows.reverse();
    if (vector[1] === 1) cols.reverse();

    let moved = false;
    let gained = 0;

    for (const r of rows){
      for (const c of cols){
        const tile = grid[r][c];
        if (!tile) continue;
        let pr = r, pc = c;          // current
        let nr = r, nc = c;          // next candidate
        // find farthest empty + possible merge target
        while (true){
          const tr = nr + vector[0];
          const tc = nc + vector[1];
          if (tr<0||tr>=SIZE||tc<0||tc>=SIZE) break;
          if (!grid[tr][tc]){ nr=tr; nc=tc; continue; }
          // occupied: merge?
          const other = grid[tr][tc];
          if (other.value === tile.value && !other.merged){
            // merge
            grid[pr][pc] = null;
            other.value *= 2;
            other.merged = true;
            grid[tr][tc] = other;
            score += other.value;
            gained += other.value;
            moved = true;
            if (other.value === WIN_VALUE && !keepPlaying) state = "won";
            pr = tr; pc = tc;
            nr = tr; nc = tc; // absorbed
          }
          break;
        }
        // slide into farthest empty (nr,nc) if changed and no merge happened at end
        if ((nr!==pr || nc!==pc) && grid[pr] && grid[pr][pc] === tile){
          grid[pr][pc] = null;
          tile.r = nr; tile.c = nc;
          grid[nr][nc] = tile;
          moved = true;
        }
      }
    }

    if (moved){
      const t = spawnTile();
      render();
      if (state === "won"){
        showOverlay(winOverlay);
      } else if (!movesAvailable()){
        state = "over";
        finalEl.textContent = score;
        showOverlay(overOverlay);
      }
    }
  }

  function movesAvailable(){
    for (let r=0;r<SIZE;r++)
      for (let c=0;c<SIZE;c++){
        if (!grid[r][c]) return true;
        const v = grid[r][c].value;
        if (r+1<SIZE && grid[r+1][c] && grid[r+1][c].value===v) return true;
        if (c+1<SIZE && grid[r][c+1] && grid[r][c+1].value===v) return true;
      }
    return false;
  }

  // ---- overlays ----
  function hideOverlays(){
    [titleOverlay,winOverlay,overOverlay].forEach(o=>o.classList.remove("show"));
  }
  function showOverlay(o){ hideOverlays(); o.classList.add("show"); }

  function newGame(){
    emptyGrid();
    score = 0;
    keepPlaying = false;
    state = "playing";
    spawnTile();
    spawnTile();
    hideOverlays();
    render();
  }

  // ---- input ----
  const keyMap = {
    ArrowLeft:"left", ArrowRight:"right", ArrowUp:"up", ArrowDown:"down",
    a:"left", d:"right", w:"up", s:"down"
  };
  document.addEventListener("keydown", (e)=>{
    if (state === "title" && (e.key === " " || e.code === "Space")){
      e.preventDefault(); newGame(); return;
    }
    const dir = keyMap[e.key];
    if (dir){ e.preventDefault(); move(dir); }
  });

  // touch / swipe
  let touchStart = null;
  boardEl.addEventListener("touchstart",(e)=>{
    if (e.touches.length===1){ touchStart = {x:e.touches[0].clientX, y:e.touches[0].clientY}; }
  }, {passive:true});
  boardEl.addEventListener("touchend",(e)=>{
    if (!touchStart) return;
    const t = e.changedTouches[0];
    const dx = t.clientX - touchStart.x;
    const dy = t.clientY - touchStart.y;
    touchStart = null;
    const absX = Math.abs(dx), absY = Math.abs(dy);
    if (Math.max(absX,absY) < 24) return;
    if (state === "title"){ return; }
    if (absX > absY) move(dx>0?"right":"left");
    else move(dy>0?"down":"up");
  }, {passive:true});

  // buttons
  document.getElementById("startBtn").addEventListener("click", newGame);
  document.getElementById("newBtn").addEventListener("click", newGame);
  document.getElementById("tryBtn").addEventListener("click", newGame);
  document.getElementById("winNewBtn").addEventListener("click", newGame);
  document.getElementById("keepBtn").addEventListener("click", ()=>{
    keepPlaying = true;
    state = "playing";
    hideOverlays();
  });

  // initial render (empty board behind title overlay)
  emptyGrid();
  render();
})();
</script>

(function () {
  "use strict";

  const SIZE = 4;
  const WIN_VALUE = 2048;
  const STORAGE_BEST = "tileMerge.best";

  const boardEl = document.getElementById("board");
  const scoreEl = document.getElementById("score");
  const bestEl = document.getElementById("best");
  const winOverlay = document.getElementById("win-overlay");
  const gameoverOverlay = document.getElementById("gameover-overlay");
  const finalScoreEl = document.getElementById("final-score");
  const titleScreen = document.getElementById("title-screen");

  let grid = [];
  let score = 0;
  let best = parseInt(localStorage.getItem(STORAGE_BEST) || "0", 10);
  let won = false;
  let keepPlayingAfterWin = false;
  let gameOver = false;
  let tileIdCounter = 1;
  let tiles = new Map(); // id -> {r,c,value,id}
  let cellSize = 0;
  let cellGap = 10;
  let padding = 10;

  bestEl.textContent = best;

  const TILE_COLORS = {
    2: ["#eee4da", "#776e65"],
    4: ["#ede0c8", "#776e65"],
    8: ["#f2b179", "#f9f6f2"],
    16: ["#f59563", "#f9f6f2"],
    32: ["#f67c5f", "#f9f6f2"],

    512: ["#edc850", "#f9f6f2"],
    1024: ["#edc53f", "#f9f6f2"],
    2048: ["#edc22e", "#f9f6f2"],
    4096: ["#3c3a32", "#f9f6f2"],
    8192: ["#1f1e1a", "#f9f6f2"]
  };

  function tileColor(v) {
    return TILE_COLORS[v] || ["#111", "#fff"];
  }

  function emptyGrid() {
    const g = [];
    for (let r = 0; r < SIZE; r++) {
      g.push(new Array(SIZE).fill(null));
    }
    return g;
  }

  function getEmptyCells() {
    const cells = [];
    for (let r = 0; r < SIZE; r++) {
      for (let c = 0; c < SIZE; c++) {
        if (!grid[r][c]) cells.push([r, c]);
      }
    }
    return cells;
  }

  function addRandomTile() {
    const empties = getEmptyCells();
    if (empties.length === 0) return;
    const [r, c] = empties[Math.floor(Math.random() * empties.length)];
    const value = Math.random() < 0.9 ? 2 : 4;
    const id = tileIdCounter++;
    const tile = { id, r, c, value, isNew: true, merged: false };
    grid[r][c] = tile;
    tiles.set(id, tile);
  }

  function buildCellBackground() {
    // remove old backgrounds
    boardEl.querySelectorAll(".cell-bg").forEach(el => el.remove());
    for (let r = 0; r < SIZE; r++) {
      for (let c = 0; c < SIZE; c++) {
        const bg = document.createElement("div");
        bg.className = "cell-bg";
        positionElement(bg, r, c);
        boardEl.appendChild(bg);
      }
    }
  }

  function computeMetrics() {
    const rect = boardEl.getBoundingClientRect();
    const totalPadding = padding * 2;
    const totalGap = cellGap * (SIZE - 1);
    cellSize = (rect.width - totalPadding - totalGap) / SIZE;
  }

  function positionElement(el, r, c) {
    const top = padding + r * (cellSize + cellGap);
    const left = padding + c * (cellSize + cellGap);
    el.style.width = cellSize + "px";
    el.style.height = cellSize + "px";
    el.style.top = top + "px";
    el.style.left = left + "px";
  }

  function newGame() {
    grid = emptyGrid();
    tiles.clear();
    score = 0;
    won = false;
    keepPlayingAfterWin = false;
    gameOver = false;
    scoreEl.textContent = "0";
    winOverlay.classList.add("hidden");
    gameoverOverlay.classList.add("hidden");
    computeMetrics();
    buildCellBackground();
    clearTileElements();
    addRandomTile();
    addRandomTile();
    render();
  }

  function clearTileElements() {
    boardEl.querySelectorAll(".tile").forEach(el => el.remove());
  }

  function render() {
    // remove DOM tiles that no longer exist
    const existingIds = new Set(tiles.keys());
    boardEl.querySelectorAll(".tile").forEach(el => {
      const id = parseInt(el.dataset.id, 10);
      if (!existingIds.has(id)) el.remove();
    });

    tiles.forEach((tile) => {
      let el = boardEl.querySelector('.tile[data-id="' + tile.id + '"]');
      if (!el) {
        el = document.createElement("div");
        el.className = "tile";
        el.dataset.id = tile.id;
        el.innerHTML = '<div class="tile-inner"></div>';
        boardEl.appendChild(el);
      }
      positionElement(el, tile.r, tile.c);
      const inner = el.querySelector(".tile-inner");
      const [bg, fg] = tileColor(tile.value);
      inner.style.background = bg;
      inner.style.color = fg;
      const fontSize = tile.value >= 1000 ? cellSize * 0.32 : cellSize * 0.42;
      inner.style.fontSize = fontSize + "px";
      inner.textContent = tile.value;
      if (tile.merged) {
        el.classList.add("merged");
      } else {
        el.classList.remove("merged");
      }
      tile.isNew = false;
      tile.merged = false;
    });

    scoreEl.textContent = score;
    if (score > best) {
      best = score;
      localStorage.setItem(STORAGE_BEST, String(best));
    }
    bestEl.textContent = best;
  }

  // returns true if any tile moved/merged
  function move(direction) {
    if (gameOver || (won && !keepPlayingAfterWin)) return;

    // direction: 'up','down','left','right'
    let moved = false;
    const vector = getVector(direction);
    const traversal = buildTraversal(vector);

    // reset merge flags
    tiles.forEach(t => { t.mergedFrom = null; });

    traversal.rows.forEach(r => {
      traversal.cols.forEach(c => {
        const tile = grid[r][c];
        if (!tile) return;
        const pos = { r, c };
        let next = findFarthestPosition(pos, vector);
        const nextCell = next.next;
        const farthest = next.farthest;

        if (nextCell && grid[nextCell.r][nextCell.c] && grid[nextCell.r][nextCell.c].value === tile.value && !grid[nextCell.r][nextCell.c].mergedFrom) {
          // merge
          const other = grid[nextCell.r][nextCell.c];
          const mergedValue = tile.value * 2;
          // move tile into other's cell then merge
          grid[r][c] = null;
          tile.r = nextCell.r;
          tile.c = nextCell.c;
          other.value = mergedValue;
          other.merged = true;
          other.mergedFrom = [tile.id, other.id];
          tiles.delete(tile.id);
          score += mergedValue;
          moved = true;
          if (mergedValue === WIN_VALUE && !won) {
            won = true;
          }
        } else {
          // move to farthest
          if (farthest.r !== r || farthest.c !== c) {
            grid[r][c] = null;
            grid[farthest.r][farthest.c] = tile;
            tile.r = farthest.r;
            tile.c = farthest.c;
            moved = true;
          }
        }
      });
    });

    if (moved) {
      addRandomTile();
      render();
      if (won && !keepPlayingAfterWin) {
        showWinOverlay();
      } else if (!canMove()) {
        gameOver = true;
        showGameOverOverlay();
      }
    }
  }

  function getVector(direction) {
    switch (direction) {
      case "up": return { r: -1, c: 0 };
      case "down": return { r: 1, c: 0 };
      case "left": return { r: 0, c: -1 };
      case "right": return { r: 0, c: 1 };
    }
  }

  function buildTraversal(vector) {
    const rows = [];
    const cols = [];
    for (let i = 0; i < SIZE; i++) { rows.push(i); cols.push(i); }
    if (vector.r === 1) rows.reverse();
    if (vector.c === 1) cols.reverse();
    return { rows, cols };
  }

  function withinBounds(r, c) {
    return r >= 0 && r < SIZE && c >= 0 && c < SIZE;
  }

  function findFarthestPosition(pos, vector) {
    let prev;
    let cur = pos;
    do {
      prev = cur;
      cur = { r: prev.r + vector.r, c: prev.c + vector.c };
    } while (withinBounds(cur.r, cur.c) && !grid[cur.r][cur.c]);
    return {
      farthest: prev,
      next: withinBounds(cur.r, cur.c) ? cur : null
    };
  }

  function canMove() {
    if (getEmptyCells().length > 0) return true;
    for (let r = 0; r < SIZE; r++) {
      for (let c = 0; c < SIZE; c++) {
        const v = grid[r][c].value;
        if (c < SIZE - 1 && grid[r][c + 1].value === v) return true;
        if (r < SIZE - 1 && grid[r + 1][c].value === v) return true;
      }
    }
    return false;
  }

  function showWinOverlay() {
    winOverlay.classList.remove("hidden");
  }

  function showGameOverOverlay() {
    finalScoreEl.textContent = "Final score: " + score;
    gameoverOverlay.classList.remove("hidden");
  }

  // Controls
  document.addEventListener("keydown", (e) => {
    if (titleScreen && !titleScreen.classList.contains("hidden")) {
      if (e.code === "Space") {
        e.preventDefault();
        startGame();
      }
      return;
    }
    const map = {
      ArrowUp: "up",
      ArrowDown: "down",
      ArrowLeft: "left",
      ArrowRight: "right",
      w: "up", s: "down", a: "left", d: "right",
      W: "up", S: "down", A: "left", D: "right"
    };
    if (map[e.key]) {
      e.preventDefault();
      move(map[e.key]);
    }
  });

  // Touch / swipe support
  let touchStartX = 0;
  let touchStartY = 0;
  boardEl.addEventListener("touchstart", (e) => {
    const t = e.changedTouches[0];
    touchStartX = t.clientX;
    touchStartY = t.clientY;
  }, { passive: true });

  boardEl.addEventListener("touchend", (e) => {
    const t = e.changedTouches[0];
    const dx = t.clientX - touchStartX;
    const dy = t.clientY - touchStartY;
    const absX = Math.abs(dx);
    const absY = Math.abs(dy);
    const threshold = 24;
    if (Math.max(absX, absY) < threshold) return;
    if (absX > absY) {
      move(dx > 0 ? "right" : "left");
    } else {
      move(dy > 0 ? "down" : "up");
    }
  }, { passive: true });

  // Buttons
  document.getElementById("new-game-btn").addEventListener("click", newGame);
  document.getElementById("tryagain-btn").addEventListener("click", newGame);
  document.getElementById("win-newgame-btn").addEventListener("click", newGame);
  document.getElementById("keep-playing-btn").addEventListener("click", () => {
    keepPlayingAfterWin = true;
    winOverlay.classList.add("hidden");
  });

  function startGame() {
    titleScreen.classList.add("hidden");
    computeMetrics();
    newGame();
  }

  document.getElementById("start-btn").addEventListener("click", startGame);
  titleScreen.addEventListener("click", (e) => {
    if (e.target === titleScreen) startGame();
  });

  window.addEventListener("resize", () => {
    computeMetrics();
    buildCellBackground();
    render();
  });

  // Initialize board metrics even before start so first render is correct
  computeMetrics();
  buildCellBackground();
})();

                    }

                        grid[r] = result.arr;
                        result.mergedIndices.forEach(c => mergedPositions.push({ r, c }));
                    }
                } else if (direction === 'right') {
                    for (let r = 0; r < GRID_SIZE; r++) {
                        const result = slideAndMerge(grid[r].slice().reverse());
                        const reversed = result.arr.reverse();
                        if (grid[r].some((v, i) => v !== reversed[i])) moved = true;
                        grid[r] = reversed;
                        result.mergedIndices.forEach(c => mergedPositions.push({ r, c: GRID_SIZE - 1 - c }));

                        }

                    }, 120);
                }
            }

            function slideAndMerge(arr) {
                let result = arr.filter(x => x !== 0);
                const mergedIndices = [];
                
                for (let i = 0; i < result.length - 1; i++) {
                    if (result[i] === result[i + 1]) {
                        result[i] *= 2;
                        result[i + 1] = 0;
                        mergedIndices.push(i);
                    }
                }
                
                result = result.filter(x => x !== 0);
                
                while (result.length < GRID_SIZE) {
                    result.push(0);
                }
                
                return { arr: result, mergedIndices };
            }

                    bestScoreEl.textContent = bestScore;

                }

                    for (let c = 0; c < GRID_SIZE; c++) {
                        if (grid[r][c] >= WIN_VALUE) return true;
                    }
                }

                // Check for empty cells

                    for (let c = 0; c < GRID_SIZE; c++) {
                        if (grid[r][c] === 0) return false;
                    }

                    for (let c = 0; c < GRID_SIZE; c++) {
                        const val = grid[r][c];
                        if (r < GRID_SIZE - 1 && grid[r + 1][c] === val) return false;
                        if (c < GRID_SIZE - 1 && grid[r][c + 1] === val) return false;
                    }
                }

                return true;
            }

            // Start
            init();
        })();

    // Grid coordinate model
    class Grid {
      constructor(size = 4) {
        this.size = size;
        this.cells = this.empty();
      }

      empty() {
        let cells = [];
        for (let x = 0; x < this.size; x++) {
          let row = cells[x] = [];
          for (let y = 0; y < this.size; y++) {
            row.push(null);
          }
        }
        return cells;
      }

      randomAvailableCell() {
        let cells = this.availableCells();
        if (cells.length) {
          return cells[Math.floor(Math.random() * cells.length)];
        }
        return null;
      }

      availableCells() {
        let cells = [];
        for (let x = 0; x < this.size; x++) {
          for (let y = 0; y < this.size; y++) {
            if (!this.cells[x][y]) {
              cells.push({ x, y });
            }
          }
        }
        return cells;
      }

      cellsAvailable() {
        return !!this.availableCells().length;
      }

      cellAvailable(cell) {
        return !this.cellOccupied(cell);
      }

      cellOccupied(cell) {
        return !!this.cellContent(cell);
      }

      cellContent(cell) {
        if (this.withinBounds(cell)) {
          return this.cells[cell.x][cell.y];
        }
        return null;
      }

      insertTile(tile) {
        this.cells[tile.x][tile.y] = tile;
      }

      removeTile(tile) {
        this.cells[tile.x][tile.y] = null;
      }

      withinBounds(cell) {
        return cell.x >= 0 && cell.x < this.size && cell.y >= 0 && cell.y < this.size;
      }
    }

    let nextTileId = 0;
    class Tile {
      constructor(x, y, value) {
        this.x = x;
        this.y = y;
        this.value = value;
        this.id = nextTileId++;
        this.previousPosition = null;
        this.mergedFrom = null;
        this.element = null;
      }

      savePosition() {
        this.previousPosition = { x: this.x, y: this.y };
      }
    }

    // Direction vector maps (0: Up, 1: Right, 2: Down, 3: Left)
    function getVector(direction) {
      const map = {
        0: { x: 0, y: -1 },
        1: { x: 1, y: 0 },
        2: { x: 0, y: 1 },
        3: { x: -1, y: 0 }
      };
      return map[direction];
    }

    function buildTraversals(vector) {
      let traversals = { x: [], y: [] };
      for (let pos = 0; pos < 4; pos++) {
        traversals.x.push(pos);
        traversals.y.push(pos);
      }

      if (vector.x === 1) traversals.x.reverse();
      if (vector.y === 1) traversals.y.reverse();

      return traversals;
    }

    class Game {
      constructor() {
        this.size = 4;
        this.bestScore = parseInt(localStorage.getItem('nexus_merge_best')) || 0;
        document.getElementById('best-score').textContent = this.bestScore;

        this.setupEventListeners();
        this.initTitleState();
      }

      initTitleState() {
        this.state = 'title';
        this.score = 0;
        this.won = false;
        this.keepPlaying = false;
        this.grid = new Grid(this.size);
        this.tilesToRemove = [];

        this.addRandomTile();
        this.render();

        document.getElementById('win-overlay').classList.remove('active');
        document.getElementById('gameover-overlay').classList.remove('active');

        document.getElementById('title-overlay').classList.remove('active');
        document.getElementById('win-overlay').classList.remove('active');
        document.getElementById('gameover-overlay').classList.remove('active');

        // Reset elements in DOM container

        this.state = 'play';
        this.score = 0;
        this.won = false;
        this.keepPlaying = false;

        this.addRandomTile();

      addRandomTile() {
        if (this.grid.cellsAvailable()) {
          let value = Math.random() < 0.9 ? 2 : 4;
          let cell = this.grid.randomAvailableCell();
          let tile = new Tile(cell.x, cell.y, value);
          this.grid.insertTile(tile);
        }
      }

      prepareTiles() {
        this.grid.cells.forEach(row => {
          row.forEach(tile => {
            if (tile) {
              tile.mergedFrom = null;
              tile.savePosition();
            }
          });
        });
      }

      moveTile(tile, cell) {
        this.grid.cells[tile.x][tile.y] = null;
        this.grid.cells[cell.x][cell.y] = tile;
        tile.x = cell.x;
        tile.y = cell.y;
      }

      move(direction) {
        if (this.state !== 'play') return;

        let vector = getVector(direction);
        let traversals = buildTraversals(vector);
        let moved = false;

        // Instantly wipe pending DOM removals from prior move to avoid ghosting
        this.tilesToRemove.forEach(tile => {
          if (tile.element) tile.element.remove();
        });
        this.tilesToRemove = [];

        this.prepareTiles();

        traversals.x.forEach(x => {
          traversals.y.forEach(y => {
            let tile = this.grid.cells[x][y];

            if (tile) {
              let positions = this.findFarthestPosition({ x, y }, vector);
              let next = this.grid.cellContent(positions.next);

              if (next && next.value === tile.value && !next.mergedFrom) {
                // Perform Merge
                let mergedValue = tile.value * 2;
                let merged = new Tile(positions.next.x, positions.next.y, mergedValue);
                merged.mergedFrom = [tile, next];

                this.grid.insertTile(merged);
                this.grid.removeTile(tile);

                // Update transit coordinate for smooth collapse slide
                tile.x = positions.next.x;
                tile.y = positions.next.y;
                this.tilesToRemove.push(tile);

                this.score += mergedValue;
                if (this.score > this.bestScore) {
                  this.bestScore = this.score;
                  localStorage.setItem('nexus_merge_best', this.bestScore);
                }

                if (mergedValue === 2048 && !this.keepPlaying) {
                  this.won = true;
                }

                moved = true;
              } else {
                if (positions.farthest.x !== tile.x || positions.farthest.y !== tile.y) {
                  this.moveTile(tile, positions.farthest);
                  moved = true;
                }
              }
            }
          });
        });

        if (moved) {
          this.addRandomTile();
          this.render();

          if (this.won && !this.keepPlaying) {
            this.state = 'win';
            setTimeout(() => {
              document.getElementById('win-overlay').classList.add('active');
            }, 300);
          } else if (this.isGameOver()) {
            this.state = 'gameover';
            document.getElementById('gameover-score').textContent = this.score;
            setTimeout(() => {
              document.getElementById('gameover-overlay').classList.add('active');
            }, 450);
          }
        }
      }

      findFarthestPosition(cell, vector) {
        let previous;
        do {
          previous = cell;
          cell = { x: previous.x + vector.x, y: previous.y + vector.y };
        } while (this.grid.withinBounds(cell) && this.grid.cellAvailable(cell));

        return {
          farthest: previous,
          next: cell
        };
      }

      isGameOver() {
        if (this.grid.cellsAvailable()) {
          return false;
        }
        for (let x = 0; x < this.size; x++) {
          for (let y = 0; y < this.size; y++) {
            let tile = this.grid.cells[x][y];
            if (tile) {
              for (let direction = 0; direction < 4; direction++) {
                let vector = getVector(direction);
                let cell = { x: x + vector.x, y: y + vector.y };
                let other = this.grid.cellContent(cell);
                if (other && other.value === tile.value) {
                  return false;
                }
              }
            }
          }
        }
        return true;
      }

        this.keepPlaying = true;

    (function () {
      'use strict';

      const SIZE = 4;
      const WIN = 2048;
      const GAP = 10;
      const ANIM_MS = 130;
      const STORAGE = 'tileMergeBest';

      const titleScreen = document.getElementById('title-screen');
      const boardEl = document.getElementById('board');
      const tilesEl = document.getElementById('tiles');
      const scoreEl = document.getElementById('score');
      const bestEl = document.getElementById('best');
      const finalScoreEl = document.getElementById('final-score');
      const winOverlay = document.getElementById('win-overlay');
      const gameOverOverlay = document.getElementById('game-over-overlay');

      let grid = [];
      let score = 0;
      let best = 0;
      let playing = false;
      let won = false;
      let keepGoing = false;
      let busy = false;
      let cellPx = 0;

      function emptyGrid() {
        return Array.from({ length: SIZE }, () => Array(SIZE).fill(0));
      }

      function loadBest() {
        best = parseInt(localStorage.getItem(STORAGE), 10) || 0;
        bestEl.textContent = best;
      }

      function saveBest() {
        if (score > best) {
          best = score;
          bestEl.textContent = best;
          localStorage.setItem(STORAGE, String(best));
        }
      }

      function measure() {
        const cell = boardEl.querySelector('.cell');
        if (cell) cellPx = cell.offsetWidth;
      }

      function tileClass(v) {
        return v <= 2048 ? 'tile-' + v : 'tile-super';
      }

      function renderTiles(opts = {}) {
        const { spawn = null, merged = [] } = opts;
        tilesEl.innerHTML = '';
        measure();

        for (let r = 0; r < SIZE; r++) {
          for (let c = 0; c < SIZE; c++) {
            const v = grid[r][c];
            if (!v) continue;

            const el = document.createElement('div');
            let cls = 'tile ' + tileClass(v);
            if (spawn && spawn.r === r && spawn.c === c) cls += ' spawn';
            if (merged.some(p => p.r === r && p.c === c)) cls += ' merge-pop';
            el.className = cls;
            el.textContent = v;
            el.style.width = cellPx + 'px';
            el.style.height = cellPx + 'px';
            el.style.top = r * (cellPx + GAP) + 'px';
            el.style.left = c * (cellPx + GAP) + 'px';
            tilesEl.appendChild(el);
          }
        }
      }

      function randomEmpty() {
        const spots = [];
        for (let r = 0; r < SIZE; r++) {
          for (let c = 0; c < SIZE; c++) {
            if (!grid[r][c]) spots.push({ r, c });
          }
        }
        if (!spots.length) return null;
        return spots[Math.floor(Math.random() * spots.length)];
      }

      function addTile() {
        const spot = randomEmpty();
        if (!spot) return null;
        grid[spot.r][spot.c] = Math.random() < 0.9 ? 2 : 4;
        return spot;
      }

      function slideLine(line) {
        const vals = line.filter(v => v !== 0);
        const out = [];
        const mergedIdx = [];
        let gained = 0;
        let i = 0;

        while (i < vals.length) {
          if (i + 1 < vals.length && vals[i] === vals[i + 1]) {
            const merged = vals[i] * 2;
            out.push(merged);
            mergedIdx.push(out.length - 1);
            gained += merged;
            i += 2;
          } else {
            out.push(vals[i]);
            i += 1;
          }
        }

        while (out.length < SIZE) out.push(0);
        return { line: out, mergedIdx, gained };
      }

      function getColumn(c) {
        const col = [];
        for (let r = 0; r < SIZE; r++) col.push(grid[r][c]);
        return col;
      }

      function setColumn(c, col) {
        for (let r = 0; r < SIZE; r++) grid[r][c] = col[r];
      }

      function move(dir) {
        if (!playing || busy) return;
        if (winOverlay.classList.contains('visible') || gameOverOverlay.classList.contains('visible')) return;

        const before = grid.map(row => row.slice());
        const mergedCells = [];
        let points = 0;

        if (dir === 'left') {
          for (let r = 0; r < SIZE; r++) {
            const res = slideLine(grid[r]);
            grid[r] = res.line;
            points += res.gained;
            res.mergedIdx.forEach(i => mergedCells.push({ r, c: i }));
          }
        } else if (dir === 'right') {
          for (let r = 0; r < SIZE; r++) {
            const res = slideLine(grid[r].slice().reverse());
            grid[r] = res.line.reverse();
            points += res.gained;
            res.mergedIdx.forEach(i => mergedCells.push({ r, c: SIZE - 1 - i }));
          }
        } else if (dir === 'up') {
          for (let c = 0; c < SIZE; c++) {
            const res = slideLine(getColumn(c));
            setColumn(c, res.line);
            points += res.gained;
            res.mergedIdx.forEach(i => mergedCells.push({ r: i, c }));
          }
        } else if (dir === 'down') {
          for (let c = 0; c < SIZE; c++) {
            const res = slideLine(getColumn(c).reverse());
            setColumn(c, res.line.reverse());
            points += res.gained;
            res.mergedIdx.forEach(i => mergedCells.push({ r: SIZE - 1 - i, c }));
          }
        }

        const changed = grid.some((row, r) => row.some((v, c) => v !== before[r][c]));
        if (!changed) return;

        busy = true;
        score += points;
        scoreEl.textContent = score;
        saveBest();
        renderTiles({ merged: mergedCells });

        window.setTimeout(() => {
          const spawn = addTile();
          renderTiles({ spawn, merged: [] });

          if (!won && !keepGoing && hasTile(WIN)) {
            won = true;
            winOverlay.classList.add('visible');
          } else if (isGameOver()) {
            finalScoreEl.textContent = score;
            gameOverOverlay.classList.add('visible');
          }

          busy = false;
        }, ANIM_MS);
      }

      function hasTile(v) {
        return grid.some(row => row.some(n => n >= v));
      }

      function isGameOver() {
        for (let r = 0; r < SIZE; r++) {
          for (let c = 0; c < SIZE; c++) {
            if (!grid[r][c]) return false;
            if (r + 1 < SIZE && grid[r + 1][c] === grid[r][c]) return false;
            if (c + 1 < SIZE && grid[r][c + 1] === grid[r][c]) return false;
          }
        }
        return true;
      }

      function newGame() {
        grid = emptyGrid();
        score = 0;
        won = false;
        keepGoing = false;
        scoreEl.textContent = '0';
        winOverlay.classList.remove('visible');
        gameOverOverlay.classList.remove('visible');
        addTile();
        addTile();
        renderTiles();
      }

      function start() {
        if (playing) return;
        playing = true;
        titleScreen.classList.add('hidden');
        newGame();
      }

      function resetFromOverlay() {
        newGame();
      }

      function onKey(e) {
        if (!playing && e.code === 'Space') {
          e.preventDefault();
          start();
          return;
        }
        if (!playing) return;

        const map = {
          ArrowUp: 'up',
          ArrowDown: 'down',
          ArrowLeft: 'left',
          ArrowRight: 'right'
        };
        if (map[e.key]) {
          e.preventDefault();
          move(map[e.key]);
        }
      }

      titleScreen.addEventListener('click', start);

      document.getElementById('keep-playing').addEventListener('click', () => {
        keepGoing = true;
        winOverlay.classList.remove('visible');
      });

      document.getElementById('new-game-win').addEventListener('click', resetFromOverlay);
      document.getElementById('try-again').addEventListener('click', resetFromOverlay);
      document.getElementById('new-game-btn').addEventListener('click', resetFromOverlay);

      let touchX = 0;
      let touchY = 0;

      boardEl.addEventListener('touchstart', e => {
        touchX = e.touches[0].clientX;
        touchY = e.touches[0].clientY;
      }, { passive: true });

      boardEl.addEventListener('touchend', e => {
        if (!playing) return;
        const dx = e.changedTouches[0].clientX - touchX;
        const dy = e.changedTouches[0].clientY - touchY;
        const min = 40;
        if (Math.abs(dx) < min && Math.abs(dy) < min) return;
        if (Math.abs(dx) > Math.abs(dy)) move(dx > 0 ? 'right' : 'left');
        else move(dy > 0 ? 'down' : 'up');
      }, { passive: true });

      window.addEventListener('keydown', onKey);
      window.addEventListener('resize', () => { if (playing) renderTiles(); });

      loadBest();
      measure();
    })();

        const result = [];

          if (!result.length) {

            tile.y = coords[0].y;

                x: last.x,
                y: last.y,
                value: last.value * 2,
                mergedThisTurn: true,
                isNew: false

              score += merged.value;
              dead.push(last.id, tile.id);
              result.pop();
              result.push(merged);
            } else {
              const i = result.length;
              tile.x = coords[i].x;
              tile.y = coords[i].y;
              result.push(tile);
            }
          }
        });
        return { tiles: result, score, dead };

            const tiles = coords.map(c => getTileAt(c.x, c.y)).filter(Boolean);
            lines.push({ tiles, coords });
          }
        }
        return lines;
      }

      function move(direction) {
        if (!state.started || state.over) return;
        const beforeGrid = gridValues();
        let nextTiles = [];
        let moveScore = 0;
        const dead = new Set();
        const lines = buildLines(direction);
        lines.forEach(line => {
          const res = processLine(line.tiles, line.coords);
          nextTiles.push(...res.tiles);
          moveScore += res.score;
          res.dead.forEach(id => dead.add(id));
        });
        const afterGrid = gridValues(nextTiles);
        if (gridsEqual(beforeGrid, afterGrid)) return;

        state.tiles = nextTiles;
        state.score += moveScore;
        if (state.score > state.best) {
          state.best = state.score;
          writeBest(state.best);
        }
        spawnTile();
        checkState();
        render();
        updateScores();
      }

      function checkState() {
        if (isGameOver()) {
          state.over = true;
          finalScoreEl.textContent = state.score;
          overScreen.classList.remove('hidden');
          return;
        }
        if (!state.won && state.tiles.some(t => t.value >= 2048)) {
          state.won = true;
          winScreen.classList.remove('hidden');
        }
      }

      function isGameOver() {
        if (state.tiles.length < SIZE * SIZE) return false;
        for (let y = 0; y < SIZE; y++) {
          for (let x = 0; x < SIZE; x++) {
            const t = getTileAt(x, y);
            const checks = [[1, 0], [0, 1]];
            for (const [dx, dy] of checks) {
              const nx = x + dx, ny = y + dy;
              if (nx < SIZE && ny < SIZE) {
                const n = getTileAt(nx, ny);
                if (n && n.value === t.value) return false;
              }
            }
          }
        }
        return true;
      }

          let el = existing[tile.id];

        });

            existing[id].remove();
          }
        });

        state.tiles.forEach(t => { t.isNew = false; t.mergedThisTurn = false; });
      }

        nextId = 1;
        state.tiles = [];
        state.score = 0;
        state.over = false;
        state.won = false;

    bestScore = Number(localStorage.getItem('bestScore')) || 0;
    bestVal.textContent = bestScore;
    resetGame();
    bindEvents();

  function resetGame() {
    grid = Array.from({length:4}, ()=>Array(4).fill(0));
    score = 0;
    won = false;
    winShown = false;
    scoreVal.textContent = score;
    hideOverlay(winOverlay);
    hideOverlay(gameOverOverlay);
    hideOverlay(titleScreen);
    showEl(gameContainer);
    spawnTile();
    spawnTile();
    render();

  function bindEvents() {
    startBtn.addEventListener('click', startGame);
    startBtn.addEventListener('keydown', e=>{ if (e.key===' ') startGame(); });
    document.addEventListener('keydown', handleKey);
    window.addEventListener('blur', ()=>{ /* ignore */ });
    // touch swipe
    let tx1, ty1;
    board.addEventListener('touchstart', e=>{ const t=e.touches[0]; tx1=t.clientX; ty1=t.clientY; }, {passive:true});
    board.addEventListener('touchend', e=>{ if (!tx1||!ty1) return; const t=e.changedTouches[0]; const dx=t.clientX-tx1; const dy=t.clientY-ty1; if (Math.abs(dx)>20||Math.abs(dy)>20) { if (Math.abs(dx)>Math.abs(dy)) { if(dx>0) handleMove('right'); else handleMove('left'); } else { if(dy>0) handleMove('down'); else handleMove('up'); } } tx1=ty1=null; }, {passive:true});
    keepPlayingBtn.addEventListener('click', ()=>{ hideOverlay(winOverlay); winShown=false; });
    newGameWinBtn.addEventListener('click', resetGame);
    tryAgainBtn.addEventListener('click', resetGame);

  function startGame() {
    hideOverlay(titleScreen);
    showEl(gameContainer);

  function spawnTile() {
    const empties = [];
    for (let r=0;r<4;r++) for (let c=0;c<4;c++) if (grid[r][c]===0) empties.push([r,c]);
    if (empties.length===0) return;
    const [r,c] = empties[Math.floor(Math.random()*empties.length)];
    grid[r][c] = Math.random()<0.9?2:4;
  }

  function canMove() {
    for (let r=0;r<4;r++) {
      for (let c=0;c<4;c++) {
        if (grid[r][c]===0) return true;
        if (r+1<4 && grid[r][c]===grid[r+1][c]) return true;
        if (c+1<4 && grid[r][c]===grid[r][c+1]) return true;
      }
    }
    return false;
  }

  function arraysEqual(a,b){ return JSON.stringify(a)===JSON.stringify(b); }

  function handleKey(e) {
    if (!gameContainer.classList.contains('hidden')) {
      let dir = null;
      switch(e.key){
        case 'ArrowUp': e.preventDefault(); dir='up'; break;
        case 'ArrowDown': e.preventDefault(); dir='down'; break;
        case 'ArrowLeft': e.preventDefault(); dir='left'; break;
        case 'ArrowRight': e.preventDefault(); dir='right'; break;
        default: return;
      }
      if (dir) handleMove(dir);
    }
  }

  function handleMove(dir) {
    const before = JSON.stringify(grid);
    let moved = false;
    if (dir==='left' || dir==='right') {
      for (let r=0;r<4;r++) {
        let row = grid[r].slice();
        if (dir==='right') row.reverse();
        // compress
        let tmp = row.filter(v=>v!==0);
        // merge
        for (let i=0;i<tmp.length-1;i++) {
          if (tmp[i]===tmp[i+1]) {
            tmp[i]*=2;
            tmp[i+1]=0;
            score+=tmp[i];
          }
        }
        // compress again
        tmp = tmp.filter(v=>v!==0);
        while(tmp.length<4) tmp.push(0);
        if (dir==='right') tmp.reverse();
        grid[r] = tmp;
      }
    } else {
      for (let c=0;c<4;c++) {
        let col = [];
        for (let r=0;r<4;r++) col.push(grid[r][c]);
        if (dir==='down') col.reverse();
        let tmp = col.filter(v=>v!==0);
        for (let i=0;i<tmp.length-1;i++) {
          if (tmp[i]===tmp[i+1]) {
            tmp[i]*=2;
            tmp[i+1]=0;
            score+=tmp[i];
          }
        }
        tmp = tmp.filter(v=>v!==0);
        while(tmp.length<4) tmp.push(0);
        if (dir==='down') tmp.reverse();
        for (let r=0;r<4;r++) grid[r][c]=tmp[r];
      }
    }
    moved = (JSON.stringify(grid)!==before);
    if (moved) {
      scoreVal.textContent = score;
      if (score>bestScore) {
        bestScore = score;
        bestVal.textContent = bestScore;
        localStorage.setItem('bestScore', bestScore);
      }
      spawnTile();
      render();
      // win detection
      if (!won && grid.some(row=>row.includes(2048))) {
        won = true;
        if (!winShown) { winShown=true; showOverlay(winOverlay); }
      }
      // game over detection
      if (!won && !canMove()) {
        showOverlay(gameOverOverlay);
        document.getElementById('final-score').textContent = score;
      }
    }
  }

  function render() {
    board.innerHTML = '';
    for (let r=0;r<4;r++) {
      for (let c=0;c<4;c++) {
        const v = grid[r][c];
        if (v===0) continue;
        const tile = document.createElement('div');
        tile.className = 'tile new';
        tile.textContent = v;
        const {bg,fg}=getColor(v);
        tile.style.backgroundColor = bg;
        tile.style.color = fg;
        tile.style.fontSize = v<100?'28px': v<1000?'22px':'18px';
        const left = c*(TILE_SIZE+GAP)+GAP;
        const top = r*(TILE_SIZE+GAP)+GAP;
        tile.style.left = `${left}px`;
        tile.style.top = `${top}px`;
        board.appendChild(tile);
      }
    }
  }

  function showEl(el){ el.classList.remove('hidden'); }
  function hideEl(el){ el.classList.add('hidden'); }
  function showOverlay(ov){ ov.classList.remove('hidden'); }
  function hideOverlay(ov){ ov.classList.add('hidden'); }

  // init on load
  window.addEventListener('DOMContentLoaded', init);
})();
