    // --- AUDIO SYSTEM (Web Audio API Retro Synth) ---
    let audioCtx = null;

    function initAudio() {
      if (audioCtx) return;
      audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    }

    function playSound(type) {
      if (!audioCtx) return;
      if (audioCtx.state === 'suspended') {
        audioCtx.resume();
      }

      const osc = audioCtx.createOscillator();
      const gain = audioCtx.createGain();
      osc.connect(gain);
      gain.connect(audioCtx.destination);

      const now = audioCtx.currentTime;

      switch(type) {
        case 'hop':
          osc.type = 'triangle';
          osc.frequency.setValueAtTime(160, now);
          osc.frequency.exponentialRampToValueAtTime(320, now + 0.12);
          gain.gain.setValueAtTime(0.18, now);
          gain.gain.exponentialRampToValueAtTime(0.01, now + 0.12);
          osc.start(now);
          osc.stop(now + 0.12);
          break;
        case 'coin':
          osc.type = 'sine';
          osc.frequency.setValueAtTime(880, now);
          gain.gain.setValueAtTime(0.15, now);
          gain.gain.exponentialRampToValueAtTime(0.05, now + 0.08);
          // Double chime
          const osc2 = audioCtx.createOscillator();
          const gain2 = audioCtx.createGain();
          osc2.connect(gain2);
          gain2.connect(audioCtx.destination);
          osc2.type = 'sine';
          osc2.frequency.setValueAtTime(1320, now + 0.08);
          gain2.gain.setValueAtTime(0.15, now + 0.08);
          gain2.gain.exponentialRampToValueAtTime(0.01, now + 0.25);
          osc2.start(now + 0.08);
          osc2.stop(now + 0.25);
          break;
        case 'death':
          osc.type = 'sawtooth';
          osc.frequency.setValueAtTime(260, now);
          osc.frequency.exponentialRampToValueAtTime(50, now + 0.4);
          gain.gain.setValueAtTime(0.3, now);
          gain.gain.exponentialRampToValueAtTime(0.01, now + 0.4);
          osc.start(now);
          osc.stop(now + 0.4);
          break;
        case 'warning':
          osc.type = 'sawtooth';
          osc.frequency.setValueAtTime(450, now);
          gain.gain.setValueAtTime(0.05, now);
          gain.gain.exponentialRampToValueAtTime(0.01, now + 0.08);
          osc.start(now);
          osc.stop(now + 0.08);
          break;
      }
    }

    function drawBlock(ctx, gx, gy, gz, w, l, h, hue, sat, light, alpha = 1) {
      const p000 = project(gx, gy, gz);
      const p100 = project(gx + w, gy, gz);
      const p010 = project(gx, gy + l, gz);
      const p110 = project(gx + w, gy + l, gz);
      const p001 = project(gx, gy, gz + h);

    function getRotatedSubBlock(sub, dir) {
      let rx = sub.rx, ry = sub.ry, rw = sub.rw, rl = sub.rl;
      if (dir === 'up') {
        return { rx, ry, rw, rl };
      } else if (dir === 'right') {
        return { rx: ry, ry: -rx, rw: rl, rl: rw };
      } else if (dir === 'down') {
        return { rx: -rx, ry: -ry, rw, rl };
      } else if (dir === 'left') {
        return { rx: -ry, ry: rx, rw: rl, rl: rw };
      }
      return { rx, ry, rw, rl };
    }

        const xOffset = rot.rx * scaleX;
        const yOffset = rot.ry * scaleY;
        const zOffset = sub.rz * scaleZ;
        const w = rot.rw * scaleX;
        const l = rot.rl * scaleY;
        const h = sub.rh * scaleZ;
        return {
          gx: gx + xOffset,
          gy: gy + yOffset,
          gz: gz + zOffset,
          w, l, h,
          hVal: sub.h, sVal: sub.s, lVal: sub.l,
          depth: (gx + xOffset) + (gy + yOffset) + (gz + zOffset) * 0.002
        };
      });

      rotatedBlocks.sort((a, b) => a.depth - b.depth);

      rotatedBlocks.forEach(b => {
        drawBlock(ctx, b.gx, b.gy, b.gz, b.w, b.l, b.h, b.hVal, b.sVal, b.lVal);
      });
    }

    // Voxel cloud generator / drawer
    function drawCloud(ctx, cloud) {
      ctx.save();
      cloud.blocks.forEach(b => {
        // Draw flat screen-space rounded clouds for sky
        ctx.fillStyle = `rgba(255, 255, 255, ${cloud.opacity * b.op})`;
        ctx.beginPath();
        ctx.roundRect(cloud.x + b.rx * cloud.scale, cloud.y + b.ry * cloud.scale, b.rw * cloud.scale, b.rh * cloud.scale, 8);
        ctx.fill();
      });
      ctx.restore();
    }

    // --- PROCEDURAL GENERATION RULES ---
    function generateLane(y) {
      if (lanes[y]) return;

      let type = 'grass';
      let count = 0;
      
      // Determine type based on weights and constraints
      if (y <= 2) {
        type = 'grass';
      } else {
        const r = Math.random();
        // Check surrounding rows to prevent continuous deadlocks
        let prevTypes = [];
        for (let i = 1; i <= 3; i++) {
          if (lanes[y - i]) prevTypes.push(lanes[y - i].type);
        }
        
        const countConsec = (t) => prevTypes.filter(x => x === t).length;

        if (r < 0.3) {
          type = 'grass';
        } else if (r < 0.65) {
          type = (countConsec('road') >= 3) ? 'grass' : 'road';
        } else if (r < 0.9) {
          type = (countConsec('river') >= 2) ? 'road' : 'river';
        } else {
          type = (countConsec('railroad') >= 1) ? 'grass' : 'railroad';
        }
      }

      const diffMultiplier = 1 + Math.min(y / 60, 1.2);
      let lane = {
        y: y,
        type: type,
        speed: 0,
        direction: Math.random() < 0.5 ? 1 : -1,
        vehicles: [],
        logs: [],
        obstacles: {}, // trees/rocks keyed by gridX
        coins: {}, // coins keyed by gridX
        trainActive: false,
        trainWarning: false,
        trainTimer: 180 + Math.random() * 200,
        trainWarningTimer: 0,
        trainX: 0,
        spawnTimer: 0,
        spawnInterval: 120
      };

      if (type === 'grass') {
        // Place obstacles (trees/rocks)
        for (let x = -7; x <= 7; x++) {
          if (x === 0 && y <= 2) continue; // Keep center clear on startup
          if (Math.random() < 0.22) {
            lane.obstacles[x] = {
              type: Math.random() < 0.7 ? 'tree' : 'rock',
              // Subtle voxel shape offset
              scaleZ: 0.8 + Math.random() * 0.4,
              colorVar: Math.floor(Math.random() * 10) - 5
            };
          } else if (Math.random() < 0.08 && x >= -5 && x <= 5) {
            // Place coin
            lane.coins[x] = {
              angle: Math.random() * Math.PI,
              bob: Math.random() * Math.PI
            };
          }
        }
      } else if (type === 'road') {
        lane.speed = (0.012 + Math.random() * 0.02) * diffMultiplier;
        lane.spawnInterval = (140 + Math.random() * 90) / diffMultiplier;
        lane.spawnTimer = Math.random() * lane.spawnInterval;
        // Pre-populate lane so it's not empty immediately
        for (let i = 0; i < 2; i++) {
          lane.vehicles.push({
            x: -8 + Math.random() * 16,
            vType: Math.random() < 0.65 ? 'car' : 'truck',
            hue: Math.random() * 360,
            sat: 80,
            light: 45
          });
        }
      } else if (type === 'river') {
        lane.speed = (0.008 + Math.random() * 0.015) * diffMultiplier;
        lane.spawnInterval = (130 + Math.random() * 70) / diffMultiplier;
        lane.spawnTimer = Math.random() * lane.spawnInterval;
        // Pre-populate logs
        for (let i = 0; i < 2; i++) {
          lane.logs.push({
            x: -8 + Math.random() * 16,
            length: 2.2 + Math.random() * 1.0
          });
        }
      } else if (type === 'railroad') {
        lane.speed = 0.32 * diffMultiplier; // FAST TRAIN
        lane.trainTimer = 180 + Math.random() * 250;
        lane.trainX = lane.direction === 1 ? -16 : 16;
        lane.signalSide = Math.random() < 0.5 ? -6 : 6;
      }

      lanes[y] = lane;
    }

      // 1. Initialize sky background clouds
      skyClouds = [];
      for (let i = 0; i < 4; i++) {
        skyClouds.push({
          x: Math.random() * 400 - 100,
          y: 40 + i * 40 + Math.random() * 20,
          speed: 0.1 + Math.random() * 0.2,
          scale: 0.8 + Math.random() * 0.8,
          opacity: 0.35 + Math.random() * 0.35,
          blocks: [
            { rx: -30, ry: -10, rw: 60, rh: 20, op: 0.7 },
            { rx: -10, ry: -20, rw: 40, rh: 25, op: 0.9 },
            { rx: 10, ry: -5, rw: 30, rh: 15, op: 0.8 }
          ]
        });
      }

      // 2. Background Horizon Hills (Blocky Parallax)
      bgHills = [];
      for (let layer = 0; layer < 3; layer++) {
        let blocks = [];
        let factor = 0.03 * (layer + 1);
        let height = 45 + layer * 25;
        let color = `hsl(285, 25%, ${28 - layer * 5}%)`;
        for (let i = -10; i < 15; i++) {
          blocks.push({
            xOffset: i * 80 + Math.random() * 30,
            width: 70 + Math.random() * 40,
            height: height + Math.random() * 30,
            color: color
          });
        }
        bgHills.push({ factor, blocks });
      }

      // Generate base map around player
      lanes = {};
      minGeneratedY = -7;
      maxGeneratedY = 15;
      for (let y = minGeneratedY; y <= maxGeneratedY; y++) {
        generateLane(y);
      }

      // Reset Player
      player.gridX = 0;
      player.gridY = 0;
      player.z = 0;
      player.targetX = 0;
      player.targetY = 0;
      player.targetZ = 0;
      player.dir = 'up';
      player.hopProgress = 0;
      player.maxReachedY = 0;
      player.dead = 0;
      player.drowning = false;
      player.blockedTimer = 0;
      player.scaleX = 1;
      player.scaleY = 1;

      document.getElementById('coin-val').innerText = '0';

    // --- GAME LOOP & UPDATES ---
    function update(dt) {
      time += dt;

      // Handle screen shake decay
      if (screenShake > 0) screenShake -= dt;

      // Clean menu updates
      if (state === 'title') {
        // Animate clouds
        skyClouds.forEach(cloud => {
          cloud.x += cloud.speed * dt;
          if (cloud.x > canvas.width + 60) cloud.x = -150;
        });
        return;
      }

      // --- PLAYER UPDATES ---
      if (player.dead > 0) {
        player.dead -= dt;
        if (player.drowning) {
          player.z -= 0.03 * dt;
        }
        if (player.dead <= 0) {
          gameOver();
        }
      } else {
        // Handle blocked bounce-back logic animation
        if (player.blockedTimer > 0) {
          player.blockedTimer -= dt;
          const v = Math.sin((player.blockedTimer / 10) * Math.PI);
          if (player.blockedDir === 'up' || player.blockedDir === 'down') {
            player.scaleY = 1 - v * 0.15;
            player.scaleZ = 1 + v * 0.08;
          } else {
            player.scaleX = 1 - v * 0.15;
            player.scaleZ = 1 + v * 0.08;
          }
        } else {
          // Normal scale lerping
          player.scaleX += (1 - player.scaleX) * 0.15 * dt;
          player.scaleY += (1 - player.scaleY) * 0.15 * dt;
          player.scaleZ += (1 - player.scaleZ) * 0.15 * dt;
        }

        // Lerp hop positions
        if (player.hopProgress > 0) {
          player.hopProgress += 0.08 * dt;
          if (player.hopProgress >= 1) {
            player.hopProgress = 0;
            player.gridX = player.targetX;
            player.gridY = player.targetY;
            player.z = 0;
            
            // Score tracking
            if (player.gridY > player.maxReachedY) {
              player.maxReachedY = player.gridY;
              score = player.maxReachedY;
              const sv = document.getElementById('score-val');
              sv.innerText = score;
              sv.classList.remove('bounce');
              void sv.offsetWidth; // Trigger DOM reflow
              sv.classList.add('bounce');
            }

            // Coin collection check
            let currentLane = lanes[Math.round(player.gridY)];
            let playerIntX = Math.round(player.gridX);
            if (currentLane && currentLane.coins[playerIntX]) {
              delete currentLane.coins[playerIntX];
              coins += 1;
              document.getElementById('coin-val').innerText = coins;
              playSound('coin');
              spawnGoldSparkles(playerIntX, player.gridY, 0.4);
            }

            // Drown landing check
            if (currentLane && currentLane.type === 'river') {
              let onLog = false;
              currentLane.logs.forEach(log => {
                const half = log.length / 2;
                if (player.gridX >= log.x - half - 0.25 && player.gridX <= log.x + half + 0.25) {
                  onLog = true;
                }
              });
              if (!onLog) {
                triggerDrowning();
              }
            }
          } else {
            // Interpolate jump trajectory
            let t = player.hopProgress;
            player.gridX = player.hopStartGridX + (player.targetX - player.hopStartGridX) * t;
            player.gridY = player.hopStartGridY + (player.targetY - player.hopStartGridY) * t;
            player.z = Math.sin(t * Math.PI) * 0.42;
            
            // Hop squash & stretch
            player.scaleZ = 1 + Math.sin(t * Math.PI) * 0.22;
            player.scaleX = 1 - Math.sin(t * Math.PI) * 0.12;
            player.scaleY = 1 - Math.sin(t * Math.PI) * 0.12;
          }
        } else {
          // Standing state - process log displacement
          let currentLane = lanes[Math.round(player.gridY)];
          if (currentLane && currentLane.type === 'river') {
            let ridingLog = null;
            currentLane.logs.forEach(log => {
              const half = log.length / 2;
              if (player.gridX >= log.x - half && player.gridX <= log.x + half) {
                ridingLog = log;
              }
            });

            if (ridingLog) {
              const drift = currentLane.speed * currentLane.direction * dt;
              player.gridX += drift;
              player.targetX = player.gridX;
              // Screen out-of-bounds check while log riding
              if (player.gridX < -6.5 || player.gridX > 6.5) {
                triggerDrowning();
              }
            } else {
              triggerDrowning();
            }
          }
        }
      }

      // --- PROCEDURAL GENERATION ADVANCEMENT ---
      const renderAheadDistance = 14;
      if (player.gridY + renderAheadDistance > maxGeneratedY) {
        // Generate new lanes ahead
        const generateTarget = Math.round(player.gridY) + renderAheadDistance;
        for (let y = maxGeneratedY + 1; y <= generateTarget; y++) {
          generateLane(y);
        }
        maxGeneratedY = generateTarget;

        // Prune old lanes far behind
        const pruneBehind = Math.round(player.gridY) - 10;
        for (let y = minGeneratedY; y < pruneBehind; y++) {
          delete lanes[y];
        }
        minGeneratedY = pruneBehind;
      }

      // Fallback screen bottom death check
      if (player.gridY < player.maxReachedY - 7 && !player.dead) {
        triggerHitDeath();
      }

      // --- CAMERA SMOOTH LERP ---
      // Targets slightly ahead of player to give foresight
      let pPos = project(player.gridX, player.gridY, player.hopProgress > 0 ? player.z : 0);
      camera.targetX = pPos.x;
      camera.targetY = pPos.y - 120;

      camera.x += (camera.targetX - camera.x) * 0.08 * dt;
      camera.y += (camera.targetY - camera.y) * 0.08 * dt;

      // --- CLOUDS AND BACKGROUND ANIM ---
      skyClouds.forEach(cloud => {
        cloud.x += cloud.speed * dt;
        if (cloud.x > canvas.width + 60) cloud.x = -150;
      });

      // --- LANES ANIMATIONS & HAZARDS ---
      for (let yKey in lanes) {
        let lane = lanes[yKey];
        const dir = lane.direction;

        if (lane.type === 'road') {
          // Update vehicles
          lane.vehicles.forEach((veh, vIdx) => {
            veh.x += lane.speed * dir * dt;
            // Recycle off-screen
            if (dir === 1 && veh.x > 9.5) veh.x = -9.5;
            if (dir === -1 && veh.x < -9.5) veh.x = 9.5;

            // Collision check
            if (!player.dead && Math.round(player.gridY) === lane.y) {
              const halfL = veh.vType === 'car' ? 0.75 : 1.1;
              if (player.gridX >= veh.x - halfL - 0.28 && player.gridX <= veh.x + halfL + 0.28) {
                // If player is in air jumping, check height collision threshold
                if (player.z < 0.3) {
                  triggerHitDeath();
                }
              }
            }
          });

          // Spawn vehicles
          lane.spawnTimer -= dt;
          if (lane.spawnTimer <= 0) {
            lane.vehicles.push({
              x: dir === 1 ? -9.5 : 9.5,
              vType: Math.random() < 0.65 ? 'car' : 'truck',
              hue: Math.random() * 360,
              sat: 80,
              light: 45
            });
            lane.spawnTimer = lane.spawnInterval * (0.85 + Math.random() * 0.3);
          }
        } else if (lane.type === 'river') {
          // Update logs
          lane.logs.forEach(log => {
            log.x += lane.speed * dir * dt;
            // Recycle logs
            if (dir === 1 && log.x > 9.5) log.x = -9.5;
            if (dir === -1 && log.x < -9.5) log.x = 9.5;
          });

          // Spawn logs
          lane.spawnTimer -= dt;
          if (lane.spawnTimer <= 0) {
            lane.logs.push({
              x: dir === 1 ? -9.5 : 9.5,
              length: 2.2 + Math.random() * 1.0
            });
            lane.spawnTimer = lane.spawnInterval * (0.85 + Math.random() * 0.3);
          }
        } else if (lane.type === 'railroad') {
          // Train approach sequencer
          lane.trainTimer -= dt;
          if (lane.trainTimer <= 0 && !lane.trainActive && !lane.trainWarning) {
            lane.trainWarning = true;
            lane.trainWarningTimer = 75; // frames warning
          }

          if (lane.trainWarning) {
            lane.trainWarningTimer -= dt;
            // Alarm sounds
            if (Math.floor(lane.trainWarningTimer) % 20 === 0) {
              playSound('warning');
            }
            if (lane.trainWarningTimer <= 0) {
              lane.trainWarning = false;
              lane.trainActive = true;
              lane.trainX = dir === 1 ? -18 : 18;
            }
          }

          if (lane.trainActive) {
            lane.trainX += lane.speed * dir * dt;
            // Collision check along train segment bounds
            if (!player.dead && Math.round(player.gridY) === lane.y) {
              // Locomotive + 3 coaches length (approx 9 grid units)
              const tLen = 9.5;
              const leftB = dir === 1 ? lane.trainX - tLen : lane.trainX;
              const rightB = dir === 1 ? lane.trainX : lane.trainX + tLen;
              if (player.gridX >= leftB - 0.2 && player.gridX <= rightB + 0.2) {
                triggerHitDeath();
              }
            }

            // Remove train once cleared
            if ((dir === 1 && lane.trainX > 18) || (dir === -1 && lane.trainX < -18)) {
              lane.trainActive = false;
              lane.trainTimer = 220 + Math.random() * 260; // Set next train interval
            }
          }
        }
      }

      // --- PARTICLES ENGINE ---
      particles.forEach((p, idx) => {
        p.x += p.vx * dt;
        p.y += p.vy * dt;
        p.z += p.vz * dt;
        p.vz -= 0.012 * dt; // Gravity
        p.rot += p.vrot * dt;
        p.life -= p.decay * dt;
        if (p.life <= 0) {
          particles.splice(idx, 1);
        }
      });
    }

    // --- DEATH HANDLING TRIGGERS ---
    function triggerHitDeath() {
      if (player.dead) return;
      player.dead = 35; // delay frames before game over
      player.drowning = false;
      playSound('death');
      screenShake = 16;
      
      // Spawn voxel chunks (explosion burst)
      for (let i = 0; i < 24; i++) {
        particles.push({
          x: player.gridX + (Math.random() - 0.5) * 0.4,
          y: player.gridY + (Math.random() - 0.5) * 0.4,
          z: player.z + 0.2 + Math.random() * 0.3,
          vx: (Math.random() - 0.5) * 0.25,
          vy: (Math.random() - 0.5) * 0.25,
          vz: 0.15 + Math.random() * 0.15,
          rot: Math.random() * Math.PI,
          vrot: (Math.random() - 0.5) * 0.3,
          size: 0.12 + Math.random() * 0.08,
          hue: activeSkin === 'chick' ? 48 : activeSkin === 'bot' ? 205 : 345,
          sat: 90,
          light: 55,
          decay: 0.035,
          life: 1
        });
      }
    }

    function triggerDrowning() {
      if (player.dead) return;
      player.dead = 30;
      player.drowning = true;
      playSound('death');

      // Splash water drop chunks
      for (let i = 0; i < 15; i++) {
        particles.push({
          x: player.gridX + (Math.random() - 0.5) * 0.4,
          y: player.gridY + (Math.random() - 0.5) * 0.4,
          z: -0.1,
          vx: (Math.random() - 0.5) * 0.12,
          vy: (Math.random() - 0.5) * 0.12,
          vz: 0.1 + Math.random() * 0.15,
          rot: Math.random() * Math.PI,
          vrot: (Math.random() - 0.5) * 0.2,
          size: 0.08 + Math.random() * 0.08,
          hue: 200,
          sat: 85,
          light: 75,
          decay: 0.05,
          life: 1
        });
      }
    }

    function spawnGoldSparkles(gx, gy, gz) {
      for (let i = 0; i < 10; i++) {
        particles.push({
          x: gx + (Math.random() - 0.5) * 0.5,
          y: gy + (Math.random() - 0.5) * 0.5,
          z: gz + (Math.random() - 0.5) * 0.3,
          vx: (Math.random() - 0.5) * 0.08,
          vy: (Math.random() - 0.5) * 0.08,
          vz: 0.05 + Math.random() * 0.08,
          rot: Math.random() * Math.PI,
          vrot: (Math.random() - 0.5) * 0.3,
          size: 0.08 + Math.random() * 0.05,
          hue: 48,
          sat: 100,
          light: 65,
          decay: 0.04,
          life: 1
        });
      }
    }

    // --- RENDER ENGINE ---
    function render() {
      // 1. CLEAR VIEWPORT
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      // Apply screen shaking offset if triggered
      ctx.save();
      if (screenShake > 0) {
        const dx = (Math.random() - 0.5) * screenShake * 0.5;
        const dy = (Math.random() - 0.5) * screenShake * 0.5;
        ctx.translate(dx, dy);

      gridX: 0, gridY: 0, z: 0,
      targetX: 0, targetY: 0,
      hopProgress: 0, hopStartX: 0, hopStartY: 0,
      dir: 'up', maxReachedY: 0,
      dead: 0, drowning: false, blockedTimer: 0, blockedDir: '',
      scaleX: 1, scaleY: 1, scaleZ: 1
    };

    let lanes = {};
    let minGeneratedY = -6;
    let maxGeneratedY = 15;
    let camera = { x: 0, y: 0, targetX: 0, targetY: 0 };
    let skyClouds = [];
    let bgHills = [];
    let particles = [];

    // --- DRAW VOXEL PRIMITIVES ---
    function drawBlock(ctx, gx, gy, gz, w, l, h, hue, sat, light, alpha = 1) {
      const p000 = project(gx, gy, gz);
      const p100 = project(gx + w, gy, gz);
      const p010 = project(gx, gy + l, gz);
      const p110 = project(gx + w, gy + l, gz);
      const p001 = project(gx, gy, gz + h);
      const p101 = project(gx + w, gy, gz + h);
      const p011 = project(gx, gy + l, gz + h);
      const p111 = project(gx + w, gy + l, gz + h);

      ctx.save();
      ctx.globalAlpha = alpha;
      ctx.strokeStyle = `rgba(0,0,0,0.12)`;
      ctx.lineWidth = 0.8;

      // Top
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${light}%)`;
      ctx.beginPath();
      ctx.moveTo(p001.x, p001.y); ctx.lineTo(p101.x, p101.y);
      ctx.lineTo(p111.x, p111.y); ctx.lineTo(p011.x, p011.y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      // Left
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${Math.max(0, light - 12)}%)`;
      ctx.beginPath();
      ctx.moveTo(p010.x, p010.y); ctx.lineTo(p110.x, p110.y);
      ctx.lineTo(p111.x, p111.y); ctx.lineTo(p011.x, p011.y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      // Right
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${Math.max(0, light - 22)}%)`;
      ctx.beginPath();
      ctx.moveTo(p100.x, p100.y); ctx.lineTo(p110.x, p110.y);
      ctx.lineTo(p111.x, p111.y); ctx.lineTo(p101.x, p101.y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      ctx.restore();
    }

    function drawShadow(ctx, gx, gy, w, l, alpha = 0.25) {
      const p00 = project(gx, gy, 0);
      const p10 = project(gx + w, gy, 0);
      const p01 = project(gx, gy + l, 0);
      const p11 = project(gx + w, gy + l, 0);
      ctx.save();
      ctx.fillStyle = `rgba(10, 4, 18, ${alpha})`;
      ctx.beginPath();
      ctx.moveTo(p00.x, p00.y); ctx.lineTo(p10.x, p10.y);
      ctx.lineTo(p11.x, p11.y); ctx.lineTo(p01.x, p01.y);
      ctx.closePath(); ctx.fill();
      ctx.restore();
    }

    function drawRotatedBlock(ctx, gx, gy, gz, w, l, h, angle, hue, sat, light) {
      const cos = Math.cos(angle);
      const sin = Math.sin(angle);
      const p = [
        {x: -w/2, y: -l/2}, {x: w/2, y: -l/2}, {x: w/2, y: l/2}, {x: -w/2, y: l/2}
      ].map(v => {
        const rx = v.x * cos - v.y * sin;
        const ry = v.x * sin + v.y * cos;
        return project(gx + rx, gy + ry, gz);
      });
      const pt = p.map(v => ({x: v.x, y: v.y - h * gridScale}));

      ctx.save();
      ctx.strokeStyle = `rgba(0,0,0,0.12)`;
      ctx.lineWidth = 0.8;

      // Top face
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${light}%)`;
      ctx.beginPath();
      ctx.moveTo(pt[0].x, pt[0].y);
      for(let i=1; i<4; i++) ctx.lineTo(pt[i].x, pt[i].y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      // Side 1
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${Math.max(0, light-12)}%)`;
      ctx.beginPath();
      ctx.moveTo(p[1].x, p[1].y); ctx.lineTo(p[2].x, p[2].y);
      ctx.lineTo(pt[2].x, pt[2].y); ctx.lineTo(pt[1].x, pt[1].y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      // Side 2
      ctx.fillStyle = `hsl(${hue}, ${sat}%, ${Math.max(0, light-22)}%)`;
      ctx.beginPath();
      ctx.moveTo(p[2].x, p[2].y); ctx.lineTo(p[3].x, p[3].y);
      ctx.lineTo(pt[3].x, pt[3].y); ctx.lineTo(pt[2].x, pt[2].y);
      ctx.closePath(); ctx.fill(); ctx.stroke();

      ctx.restore();
    }

    // --- PROCEDURAL GENERATION ---
    function generateLane(y) {
      if (lanes[y]) return;
      let type = 'grass';
      if (y > 2) {
        let prev = [];
        for (let i = 1; i <= 3; i++) {
          if (lanes[y-i]) prev.push(lanes[y-i].type);
        }
        const count = t => prev.filter(x => x === t).length;
        const r = Math.random();
        if (r < 0.28) type = 'grass';
        else if (r < 0.62) type = count('road') >= 3 ? 'grass' : 'road';
        else if (r < 0.88) type = count('river') >= 2 ? 'road' : 'river';
        else type = count('railroad') >= 1 ? 'grass' : 'railroad';
      }

      const diff = 1 + Math.min(y / 65, 1.2);
      let lane = {
        y, type,
        speed: 0,
        direction: Math.random() < 0.5 ? 1 : -1,
        vehicles: [], logs: [], obstacles: {}, coins: {},
        trainActive: false, trainWarning: false,
        trainTimer: 150 + Math.random() * 200,
        trainWarningTimer: 0, trainX: 0,
        spawnTimer: 0, spawnInterval: 120
      };

      if (type === 'grass') {
        for (let x = -7; x <= 7; x++) {
          if (x === 0 && y <= 2) continue;
          if (Math.random() < 0.2) {
            lane.obstacles[x] = {
              vType: Math.random() < 0.75 ? 'tree' : 'rock',
              scaleZ: 0.85 + Math.random() * 0.35,
              cVar: Math.floor(Math.random() * 10) - 5
            };
          } else if (Math.random() < 0.08 && x >= -5 && x <= 5) {
            lane.coins[x] = { angle: Math.random() * Math.PI, bob: Math.random() * Math.PI };
          }
        }
      } else if (type === 'road') {
        lane.speed = (0.01 + Math.random() * 0.02) * diff;
        lane.spawnInterval = (130 + Math.random() * 90) / diff;
        lane.spawnTimer = Math.random() * lane.spawnInterval;
        for (let i = 0; i < 2; i++) {
          lane.vehicles.push({
            x: -8 + Math.random() * 16,
            vType: Math.random() < 0.7 ? 'car' : 'truck',
            hue: Math.random() * 360,
            sat: 75, light: 45
          });
        }
      } else if (type === 'river') {
        lane.speed = (0.007 + Math.random() * 0.013) * diff;
        lane.spawnInterval = (120 + Math.random() * 70) / diff;
        lane.spawnTimer = Math.random() * lane.spawnInterval;
        for (let i = 0; i < 2; i++) {
          lane.logs.push({ x: -8 + Math.random() * 16, length: 2.2 + Math.random() * 0.8 });
        }
      } else if (type === 'railroad') {
        lane.speed = 0.32 * diff;
        lane.trainTimer = 160 + Math.random() * 220;
        lane.trainX = lane.direction === 1 ? -16 : 16;
        lane.signalSide = Math.random() < 0.5 ? -6 : 6;
      }
      lanes[y] = lane;
    }

    // --- SETUP / INITIALIZE ---
    function init() {
      skyClouds = [];
      for (let i = 0; i < 4; i++) {
        skyClouds.push({
          x: Math.random() * 350 - 50,
          y: 40 + i * 35 + Math.random() * 15,
          speed: 0.12 + Math.random() * 0.15,
          scale: 0.8 + Math.random() * 0.6,
          opacity: 0.4 + Math.random() * 0.3,
          blocks: [
            { rx: -24, ry: -8, rw: 48, rh: 16, op: 0.75 },
            { rx: -8, ry: -16, rw: 32, rh: 20, op: 0.9 },
            { rx: 8, ry: -4, rw: 24, rh: 12, op: 0.8 }
          ]
        });
      }

      bgHills = [];
      for (let layer = 0; layer < 3; layer++) {
        let blocks = [];
        let factor = 0.035 * (layer + 1);
        let height = 40 + layer * 22;
        let color = `hsl(285, 25%, ${26 - layer * 4}%)`;
        for (let i = -8; i < 12; i++) {
          blocks.push({
            xOffset: i * 75 + Math.random() * 25,
            width: 60 + Math.random() * 40,
            height: height + Math.random() * 25,
            color
          });
        }
        bgHills.push({ factor, blocks });
      }

      lanes = {};
      minGeneratedY = -6;
      maxGeneratedY = 15;
      for (let y = minGeneratedY; y <= maxGeneratedY; y++) generateLane(y);

      player.gridX = 0; player.gridY = 0; player.z = 0;
      player.targetX = 0; player.targetY = 0;
      player.dir = 'up'; player.hopProgress = 0; player.maxReachedY = 0;
      player.dead = 0; player.drowning = false; player.blockedTimer = 0;
      player.scaleX = 1; player.scaleY = 1; player.scaleZ = 1;

      score = 0; coins = 0;
      document.getElementById('score-val').innerText = '0';
      document.getElementById('coin-val').innerText = '0';
      particles = [];
      screenShake = 0;

      let pPos = project(player.gridX, player.gridY, 0);
      camera.x = pPos.x; camera.y = pPos.y - 100;
    }

    // --- MAIN TICK LOOP ---
    function update(dt) {
      time += dt;
      if (screenShake > 0) screenShake -= dt;

      if (state === 'title') {
        skyClouds.forEach(c => {
          c.x += c.speed * dt;
          if (c.x > canvas.width + 50) c.x = -120;
        });
        return;
      }

      // Player State Machine
      if (player.dead > 0) {
        player.dead -= dt;
        if (player.drowning) player.z -= 0.025 * dt;
        if (player.dead <= 0) triggerGameOver();
      } else {
        if (player.blockedTimer > 0) {
          player.blockedTimer -= dt;
          const v = Math.sin((player.blockedTimer / 10) * Math.PI);
          if (player.blockedDir === 'up' || player.blockedDir === 'down') {
            player.scaleY = 1 - v * 0.15; player.scaleZ = 1 + v * 0.08;
          } else {
            player.scaleX = 1 - v * 0.15; player.scaleZ = 1 + v * 0.08;
          }
        } else {
          player.scaleX += (1 - player.scaleX) * 0.15 * dt;
          player.scaleY += (1 - player.scaleY) * 0.15 * dt;
          player.scaleZ += (1 - player.scaleZ) * 0.15 * dt;
        }

        if (player.hopProgress > 0) {
          player.hopProgress += 0.085 * dt;
          if (player.hopProgress >= 1) {
            player.hopProgress = 0;
            player.gridX = player.targetX;
            player.gridY = player.targetY;
            player.z = 0;

            if (player.gridY > player.maxReachedY) {
              player.maxReachedY = player.gridY;
              score = player.maxReachedY;
              document.getElementById('score-val').innerText = score;
            }

            let cur = lanes[Math.round(player.gridY)];
            let px = Math.round(player.gridX);
            if (cur && cur.coins[px]) {
              delete cur.coins[px];
              coins++;
              document.getElementById('coin-val').innerText = coins;
              playSound('coin');
              spawnCoinBurst(px, player.gridY);
            }

            if (cur && cur.type === 'river') {
              let onLog = false;
              cur.logs.forEach(log => {
                const half = log.length / 2;
                if (player.gridX >= log.x - half - 0.2 && player.gridX <= log.x + half + 0.2) onLog = true;
              });
              if (!onLog) killPlayer(true);
            }
          } else {
            let t = player.hopProgress;
            player.gridX = player.hopStartX + (player.targetX - player.hopStartX) * t;
            player.gridY = player.hopStartY + (player.targetY - player.hopStartY) * t;
            player.z = Math.sin(t * Math.PI) * 0.4;
            player.scaleZ = 1 + Math.sin(t * Math.PI) * 0.2;
            player.scaleX = player.scaleY = 1 - Math.sin(t * Math.PI) * 0.1;
          }
        } else {
          let cur = lanes[Math.round(player.gridY)];
          if (cur && cur.type === 'river') {
            let log = cur.logs.find(l => player.gridX >= l.x - l.length/2 && player.gridX <= l.x + l.length/2);
            if (log) {
              const drift = cur.speed * cur.direction * dt;
              player.gridX += drift;
              player.targetX = player.gridX;
              if (player.gridX < -6.5 || player.gridX > 6.5) killPlayer(true);
            } else {
              killPlayer(true);
            }
          }
        }
      }

      // Camera Lerp
      let pPos = project(player.gridX, player.gridY, player.hopProgress > 0 ? player.z : 0);
      camera.targetX = pPos.x;
      camera.targetY = pPos.y - 110;
      camera.x += (camera.targetX - camera.x) * 0.08 * dt;
      camera.y += (camera.targetY - camera.y) * 0.08 * dt;

      // Map limits death
      if (player.gridY < player.maxReachedY - 7 && !player.dead) killPlayer(false);

      // Procedural Generation bounds
      if (player.gridY + 12 > maxGeneratedY) {
        const target = Math.round(player.gridY) + 12;
        for (let y = maxGeneratedY + 1; y <= target; y++) generateLane(y);
        maxGeneratedY = target;
        const prune = Math.round(player.gridY) - 8;
        for (let y = minGeneratedY; y < prune; y++) delete lanes[y];
        minGeneratedY = prune;
      }

      skyClouds.forEach(c => {
        c.x += c.speed * dt;
        if (c.x > canvas.width + 50) c.x = -120;
      });

      // Update Hazards & Roads
      for (let yKey in lanes) {
        let lane = lanes[yKey];
        const dir = lane.direction;

        if (lane.type === 'road') {
          lane.vehicles.forEach(veh => {
            veh.x += lane.speed * dir * dt;
            if (dir === 1 && veh.x > 9) veh.x = -9;
            if (dir === -1 && veh.x < -9) veh.x = 9;

            if (!player.dead && Math.round(player.gridY) === lane.y) {
              const len = veh.vType === 'car' ? 0.7 : 1.1;
              if (player.gridX >= veh.x - len - 0.25 && player.gridX <= veh.x + len + 0.25 && player.z < 0.28) {
                killPlayer(false);
              }
            }
          });

          lane.spawnTimer -= dt;
          if (lane.spawnTimer <= 0) {
            lane.vehicles.push({
              x: dir === 1 ? -9 : 9,
              vType: Math.random() < 0.65 ? 'car' : 'truck',
              hue: Math.random() * 360, sat: 75, light: 45
            });
            lane.spawnTimer = lane.spawnInterval * (0.8 + Math.random() * 0.4);
          }
        } else if (lane.type === 'river') {
          lane.logs.forEach(log => {
            log.x += lane.speed * dir * dt;
            if (dir === 1 && log.x > 9) log.x = -9;
            if (dir === -1 && log.x < -9) log.x = 9;
          });

          lane.spawnTimer -= dt;
          if (lane.spawnTimer <= 0) {
            lane.logs.push({ x: dir === 1 ? -9 : 9, length: 2.2 + Math.random() * 0.8 });
            lane.spawnTimer = lane.spawnInterval * (0.8 + Math.random() * 0.4);
          }
        } else if (lane.type === 'railroad') {
          lane.trainTimer -= dt;
          if (lane.trainTimer <= 0 && !lane.trainActive && !lane.trainWarning) {
            lane.trainWarning = true;
            lane.trainWarningTimer = 70;
          }

          if (lane.trainWarning) {
            lane.trainWarningTimer -= dt;
            if (Math.floor(lane.trainWarningTimer) % 20 === 0) playSound('warn');
            if (lane.trainWarningTimer <= 0) {
              lane.trainWarning = false;
              lane.trainActive = true;
              lane.trainX = dir === 1 ? -16 : 16;
            }
          }

          if (lane.trainActive) {
            lane.trainX += lane.speed * dir * dt;
            if (!player.dead && Math.round(player.gridY) === lane.y) {
              const tLen = 9;
              const left = dir === 1 ? lane.trainX - tLen : lane.trainX;
              const right = dir === 1 ? lane.trainX : lane.trainX + tLen;
              if (player.gridX >= left - 0.2 && player.gridX <= right + 0.2) killPlayer(false);
            }
            if ((dir === 1 && lane.trainX > 16) || (dir === -1 && lane.trainX < -16)) {
              lane.trainActive = false;
              lane.trainTimer = 180 + Math.random() * 240;
            }
          }
        }
      }

      // Particles Engine
      particles.forEach((p, idx) => {
        p.x += p.vx * dt; p.y += p.vy * dt; p.z += p.vz * dt;
        p.vz -= 0.015 * dt; p.rot += p.vrot * dt; p.life -= p.decay * dt;
        if (p.life <= 0) particles.splice(idx, 1);
      });
    }

    function killPlayer(drowning = false) {
      if (player.dead) return;
      player.dead = 30;
      player.drowning = drowning;
      playSound('death');
      screenShake = 12;

      const pColor = activeSkin === 'chick' ? 48 : activeSkin === 'bot' ? 205 : 345;
      if (!drowning) {
        for (let i = 0; i < 18; i++) {
          particles.push({
            x: player.gridX + (Math.random() - 0.5) * 0.3,
            y: player.gridY + (Math.random() - 0.5) * 0.3,
            z: player.z + 0.2 + Math.random() * 0.2,
            vx: (Math.random() - 0.5) * 0.2,
            vy: (Math.random() - 0.5) * 0.2,
            vz: 0.12 + Math.random() * 0.12,
            rot: Math.random() * Math.PI,
            vrot: (Math.random() - 0.5) * 0.2,
            size: 0.1 + Math.random() * 0.06,
            hue: pColor, sat: 90, light: 55, decay: 0.04, life: 1
          });
        }
      } else {
        for (let i = 0; i < 12; i++) {
          particles.push({
            x: player.gridX + (Math.random() - 0.5) * 0.3,
            y: player.gridY + (Math.random() - 0.5) * 0.3,
            z: -0.1,
            vx: (Math.random() - 0.5) * 0.08,
            vy: (Math.random() - 0.5) * 0.08,
            vz: 0.08 + Math.random() * 0.12,
            rot: Math.random() * Math.PI, vrot: 0.1,
            size: 0.07 + Math.random() * 0.07,
            hue: 200, sat: 80, light: 75, decay: 0.05, life: 1
          });
        }
      }
    }

    function spawnCoinBurst(gx, gy) {
      for (let i = 0; i < 8; i++) {
        particles.push({
          x: gx + (Math.random() - 0.5) * 0.4,
          y: gy + (Math.random() - 0.5) * 0.4,
          z: 0.3,
          vx: (Math.random() - 0.5) * 0.06,
          vy: (Math.random() - 0.5) * 0.06,
          vz: 0.06 + Math.random() * 0.06,
          rot: Math.random() * Math.PI, vrot: 0.2,
          size: 0.06 + Math.random() * 0.04,
          hue: 48, sat: 100, light: 65, decay: 0.05, life: 1
        });
      }
    }

    // --- RENDERING WORK ---
    function render() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      ctx.save();
      if (screenShake > 0) {
        ctx.translate((Math.random() - 0.5) * screenShake * 0.5, (Math.random() - 0.5) * screenShake * 0.5);
      }

      // Draw Sky
      const skyGrad = ctx.createLinearGradient(0, 0, 0, canvas.height * 0.42);
      skyGrad.addColorStop(0, '#0a0518');
      skyGrad.addColorStop(0.4, '#1b103c');
      skyGrad.addColorStop(0.7, '#4e286c');
      skyGrad.addColorStop(1, '#ff7a70');
      ctx.fillStyle = skyGrad;
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      // Draw Parallax Horizon Hills
      bgHills.forEach(layer => {
        ctx.fillStyle = layer.blocks[0].color;
        ctx.beginPath();
        layer.blocks.forEach(block => {
          let sx = (block.xOffset - camera.x * layer.factor) % (canvas.width + block.width * 2);
          if (sx < -block.width) sx += (canvas.width + block.width * 2);
          ctx.rect(sx, canvas.height * 0.42 - block.height, block.width, block.height);
        });
        ctx.fill();
      });

      // Clouds
      skyClouds.forEach(c => drawCloud(ctx, c));

      // Translate context to Camera coordinates
      ctx.save();
      ctx.translate(canvas.width / 2, canvas.height / 2);
      ctx.translate(-camera.x, -camera.y);

      const queue = [];
      const currentIntY = Math.round(player.gridY);
      const startY = currentIntY - 7;
      const endY = currentIntY + 11;

      for (let y = startY; y <= endY; y++) {
        const lane = lanes[y];
        if (!lane) continue;

        // Ground block tiles
        for (let x = -8; x <= 8; x++) {
          let hue = 0, sat = 0, light = 0, h = 0.2, gz = -0.2;
          if (lane.type === 'grass') {
            hue = HSL_GRASS.h; sat = HSL_GRASS.s;
            light = HSL_GRASS.l + ((x + y) % 3 === 0 ? 3 : -2);
          } else if (lane.type === 'road') {
            hue = HSL_ROAD.h; sat = HSL_ROAD.s; light = HSL_ROAD.l;
            h = 0.18; gz = -0.22;
          } else if (lane.type === 'river') {
            hue = HSL_RIVER.h; sat = HSL_RIVER.s;
            light = HSL_RIVER.l + (Math.floor(time * 0.04 + x) % 2 === 0 ? 3 : 0);
            h = 0.08; gz = -0.3;
          } else if (lane.type === 'railroad') {
            hue = HSL_RAIL.h; sat = HSL_RAIL.s; light = HSL_RAIL.l;
            h = 0.18; gz = -0.22;
          }

          queue.push({
            depth: (x + y) * 10,
            draw: () => drawBlock(ctx, x - 0.5, y - 0.5, gz, 1, 1, h, hue, sat, light)
          });

          // Dashed road markers
          if (lane.type === 'road' && x % 2 === 0) {
            queue.push({
              depth: (x + y) * 10 + 1,
              draw: () => drawBlock(ctx, x - 0.25, y - 0.04, -0.04, 0.5, 0.08, 0.002, 45, 95, 75)
            });
          }

          // Rails sleeper lines
          if (lane.type === 'railroad') {
            if (x >= -7 && x <= 7) {
              queue.push({
                depth: (x + y) * 10 + 1,
                draw: () => drawBlock(ctx, x - 0.1, y - 0.45, -0.04, 0.2, 0.9, 0.02, 20, 30, 20)
              });
            }
            queue.push({
              depth: (x + y) * 10 + 2,
              draw: () => {
                drawBlock(ctx, x - 0.5, y - 0.24, -0.02, 1.0, 0.06, 0.04, 0, 0, 65);
                drawBlock(ctx, x - 0.5, y + 0.18, -0.02, 1.0, 0.06, 0.04, 0, 0, 65);
              }
            });
          }
        }

        // Trees & rocks obstacles
        if (lane.type === 'grass') {
          for (let xKey in lane.obstacles) {
            const obs = lane.obstacles[xKey];
            const gx = parseInt(xKey);
            queue.push({
              depth: (gx + lane.y) * 10 + 3,
              draw: () => {
                drawShadow(ctx, gx - 0.28, lane.y - 0.28, 0.56, 0.56);
                if (obs.type === 'tree') {
                  drawBlock(ctx, gx - 0.08, lane.y - 0.08, 0, 0.16, 0.16, 0.45, 25, 45, 25 + obs.cVar);
                  drawBlock(ctx, gx - 0.35, lane.y - 0.35, 0.45, 0.7, 0.7, 0.45 * obs.scaleZ, 135, 50, 38 + obs.cVar);
                  drawBlock(ctx, gx - 0.2, lane.y - 0.2, 0.45 + 0.45 * obs.scaleZ, 0.4, 0.4, 0.35 * obs.scaleZ, 135, 55, 46 + obs.cVar);
                } else {
                  drawBlock(ctx, gx - 0.22, lane.y - 0.22, 0, 0.44, 0.44, 0.28 * obs.scaleZ, 0, 0, 52 + obs.cVar);
                  drawBlock(ctx, gx - 0.1, lane.y - 0.1, 0.28 * obs.scaleZ, 0.2, 0.2, 0.12 * obs.scaleZ, 0, 0, 66 + obs.cVar);
                }
              }
            });
          }

          // Safe ground Coins
          for (let xKey in lane.coins) {
            const cn = lane.coins[xKey];
            const gx = parseInt(xKey);
            cn.angle += 0.04; cn.bob += 0.06;
            const floatZ = 0.2 + Math.sin(cn.bob) * 0.05;
            
            queue.push({
              depth: (gx + lane.y) * 10 + 4,
              draw: () => {
                drawShadow(ctx, gx - 0.16, lane.y - 0.16, 0.32, 0.32, 0.15);
                drawRotatedBlock(ctx, gx, lane.y, floatZ, 0.28 * Math.sin(cn.angle), 0.28, 0.28, 0, 48, 100, 58);
              }
            });
          }
        }

        // Vehicles
        if (lane.type === 'road') {
          lane.vehicles.forEach(veh => {
            queue.push({
              depth: (veh.x + lane.y) * 10 + 3,
              draw: () => {
                const dir = lane.direction;
                if (veh.vType === 'car') {
                  drawShadow(ctx, veh.x - 0.65, lane.y - 0.28, 1.3, 0.56);
                  // Wheels
                  drawBlock(ctx, veh.x + (dir * 0.35) - 0.12, lane.y - 0.32, 0, 0.24, 0.06, 0.16, 0, 0, 10);
                  drawBlock(ctx, veh.x + (dir * 0.35) - 0.12, lane.y + 0.26, 0, 0.24, 0.06, 0.16, 0, 0, 10);
                  drawBlock(ctx, veh.x - (dir * 0.35) - 0.12, lane.y - 0.32, 0, 0.24, 0.06, 0.16, 0, 0, 10);
                  drawBlock(ctx, veh.x - (dir * 0.35) - 0.12, lane.y + 0.26, 0, 0.24, 0.06, 0.16, 0, 0, 10);
                  // Body
                  drawBlock(ctx, veh.x - 0.65, lane.y - 0.28, 0.08, 1.3, 0.56, 0.28, veh.hue, veh.sat, veh.light);
                  drawBlock(ctx, veh.x - (dir * 0.18) - 0.3, lane.y - 0.22, 0.36, 0.6, 0.44, 0.26, veh.hue, veh.sat, veh.light);
                  drawBlock(ctx, veh.x + (dir * 0.26) - 0.02, lane.y - 0.18, 0.38, 0.04, 0.36, 0.18, 190, 85, 75); // glass
                } else {
                  // Truck
                  drawShadow(ctx, veh.x - 1.0, lane.y - 0.3, 2.0, 0.6);
                  for (let wx = -0.6; wx <= 0.6; wx += 0.6) {
                    drawBlock(ctx, veh.x + wx - 0.12, lane.y - 0.34, 0, 0.24, 0.06, 0.18, 0, 0, 10);
                    drawBlock(ctx, veh.x + wx - 0.12, lane.y + 0.28, 0, 0.24, 0.06, 0.18, 0, 0, 10);
                  }
                  drawBlock(ctx, veh.x + (dir * 0.45) - 0.3, lane.y - 0.26, 0.08, 0.55, 0.52, 0.6, veh.hue, veh.sat, veh.light); // cab
                  drawBlock(ctx, veh.x - (dir * 0.25) - 0.75, lane.y - 0.28, 0.08, 1.5, 0.56, 0.8, 0, 0, 80); // trailer
                }
              }
            });
          });
        }

        // River Logs
        if (lane.type === 'river') {
          lane.logs.forEach(log => {
            queue.push({
              depth: (log.x + lane.y) * 10 + 2,
              draw: () => {
                drawBlock(ctx, log.x - log.length/2, lane.y - 0.3, -0.06, log.length, 0.6, 0.15, 25, 45, 28);
                // Wood strip texture
                drawBlock(ctx, log.x - 0.5, lane.y - 0.15, 0.09, 0.8, 0.1, 0.01, 25, 45, 20);
              }
            });
          });
        }

        // Railroad Lights & Signal Pole
        if (lane.type === 'railroad') {
          const sigX = lane.signalSide;
          queue.push({
            depth: (sigX + lane.y) * 10 + 3,
            draw: () => {
              drawShadow(ctx, sigX - 0.12, lane.y - 0.12, 0.24, 0.24);
              drawBlock(ctx, sigX - 0.05, lane.y - 0.05, 0, 0.1, 0.1, 0.95, 0, 0, 45); // pole
              drawBlock(ctx, sigX - 0.2, lane.y - 0.1, 0.95, 0.4, 0.2, 0.25, 0, 0, 15); // box
              // Dual flashing lights
              let isLeft = Math.floor(time * 0.12) % 2 === 0;
              let lCol = (lane.trainWarning && isLeft) ? 60 : 25;
              let rCol = (lane.trainWarning && !isLeft) ? 60 : 25;
              drawBlock(ctx, sigX - 0.15, lane.y + 0.11, 1.0, 0.1, 0.02, 0.12, 0, 100, lCol);
              drawBlock(ctx, sigX + 0.05, lane.y + 0.11, 1.0, 0.1, 0.02, 0.12, 0, 100, rCol);
            }
          });

          // Zooming Train
          if (lane.trainActive) {
            const dir = lane.direction;
            const tLen = 9.0;
            queue.push({
              depth: (lane.trainX + lane.y) * 10 + 4,
              draw: () => {
                // Train Locomotive Engine Box
                const locX = lane.trainX - (dir === 1 ? 0 : tLen);
                drawShadow(ctx, locX, lane.y - 0.3, tLen, 0.6);
                
                // Draw 5 continuous long train blocks
                for (let i = 0; i < 5; i++) {
                  let chunkX = lane.trainX - (dir === 1 ? i * 1.8 : (tLen - i * 1.8));
                  let isLoc = i === 0;
                  let color = isLoc ? {h: 0, s: 80, l: 45} : {h: 210, s: 20, l: 65};
                  drawBlock(ctx, chunkX - 0.85, lane.y - 0.28, 0.08, 1.7, 0.56, 0.8, color.h, color.s, color.l);
                  // Windows
                  drawBlock(ctx, chunkX - 0.6, lane.y - 0.3, 0.42, 1.2, 0.02, 0.2, 0, 0, 12);
                }
              }
            });
          }
        }
      }

      // Add Player character to queue
      if (!player.dead || player.drowning) {
        queue.push({
          depth: (player.gridX + player.gridY) * 10 + 3.5,
          draw: () => {
            if (!player.drowning) drawShadow(ctx, player.gridX - 0.22, player.gridY - 0.22, 0.44, 0.44);
            drawCharacter(ctx, activeSkin, player.gridX, player.gridY, player.z, player.dir, player.scaleX, player.scaleY, player.scaleZ);
          }
        });
      }

      // Add Particles
      particles.forEach(p => {
        queue.push({
          depth: (p.x + p.y) * 10 + 5,
          draw: () => drawBlock(ctx, p.x - p.size/2, p.y - p.size/2, p.z, p.size, p.size, p.size, p.hue, p.sat, p.light, p.life)
        });
      });

      // Sort and draw isometric elements
      queue.sort((a, b) => a.depth - b.depth);
      queue.forEach(item => item.draw());

      ctx.restore(); // Camera
      ctx.restore(); // Shake
    }

    // --- INPUT CONTROLLER HANDLERS ---
    const keys = {};
    window.addEventListener('keydown', e => {
      if (['Space', 'ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight', 'KeyW', 'KeyS', 'KeyA', 'KeyD'].includes(e.code)) {
        e.preventDefault();
      }
      keys[e.code] = true;
      initAudio();

      if (state === 'playing') {
        if (player.dead) return;
        if (player.hopProgress === 0) {
          let dx = 0, dy = 0, newDir = player.dir;
          if (e.code === 'ArrowUp' || e.code === 'KeyW') { dy = 1; newDir = 'up'; }
          else if (e.code === 'ArrowDown' || e.code === 'KeyS') { dy = -1; newDir = 'down'; }
          else if (e.code === 'ArrowLeft' || e.code === 'KeyA') { dx = -1; newDir = 'left'; }
          else if (e.code === 'ArrowRight' || e.code === 'KeyD') { dx = 1; newDir = 'right'; }

          if (dx !== 0 || dy !== 0) triggerMove(dx, dy, newDir);
        }
      } else if (state === 'title' && e.code === 'Space') {
        startGame();
      } else if (state === 'gameover' && e.code === 'Space') {
        restartGame();
      }
    });

    window.addEventListener('keyup', e => { keys[e.code] = false; });

    // Touch Controls swipe gestures
    let touchStartX = 0, touchStartY = 0;
    canvas.addEventListener('touchstart', e => {
      touchStartX = e.touches[0].clientX;
      touchStartY = e.touches[0].clientY;
      initAudio();
    }, { passive: true });

    canvas.addEventListener('touchend', e => {
      if (state !== 'playing' || player.dead) return;
      const dx = e.changedTouches[0].clientX - touchStartX;
      const dy = e.changedTouches[0].clientY - touchStartY;
      const dist = Math.hypot(dx, dy);

      if (dist > 25) {
        let hopX = 0, hopY = 0, nd = player.dir;
        if (Math.abs(dx) > Math.abs(dy)) {
          if (dx > 0) { hopX = 1; nd = 'right'; }
          else { hopX = -1; nd = 'left'; }
        } else {
          if (dy < 0) { hopY = 1; nd = 'up'; }
          else { hopY = -1; nd = 'down'; }
        }
        triggerMove(hopX, hopY, nd);
      } else {
        // Tapping screen jumps player forward
        triggerMove(0, 1, 'up');
      }
    });

    function triggerMove(dx, dy, newDir) {
      player.dir = newDir;
      let tx = Math.round(player.gridX) + dx;
      let ty = Math.round(player.gridY) + dy;

      if (tx < -7 || tx > 7) {
        triggerBlock(newDir);
        return;
      }

      let nextLane = lanes[ty];
      if (nextLane && nextLane.type === 'grass' && nextLane.obstacles[tx]) {
        triggerBlock(newDir);
        return;
      }

      player.targetX = tx;
      player.targetY = ty;
      player.hopStartX = player.gridX;
      player.hopStartY = player.gridY;
      player.hopProgress = 0.01;
      playSound('hop');
    }

    function triggerBlock(dir) {
      player.blockedTimer = 10;
      player.blockedDir = dir;
      playSound('hop'); // Slight thud sound
    }

    // --- GAME STATES CONTROLLER ---
    function startGame() {
      initAudio();
      document.getElementById('title-screen').classList.remove('active');
      document.getElementById('hud').classList.add('active');
      init();
      state = 'playing';
    }

    function triggerGameOver() {
      state = 'gameover';
      document.getElementById('hud').classList.remove('active');
      document.getElementById('end-score').innerText = score;
      if (score > bestScore) {
        bestScore = score;
        localStorage.setItem('voxel_hop_best_score', bestScore);
      }
      document.getElementById('end-best').innerText = bestScore;
      document.getElementById('gameover-screen').classList.add('active');
    }

    function restartGame() {
      document.getElementById('gameover-screen').classList.remove('active');
      document.getElementById('hud').classList.add('active');
      init();
      state = 'playing';
      playSound('hop');
    }

    // --- CLOUD RENDER DRAW ---
    function drawCloud(ctx, cloud) {
      ctx.save();
      cloud.blocks.forEach(b => {
        ctx.fillStyle = `rgba(255, 255, 255, ${cloud.opacity * b.op})`;
        ctx.beginPath();
        // Screen space drawing of puffy voxel shape clouds
        ctx.roundRect(cloud.x + b.rx * cloud.scale, cloud.y + b.ry * cloud.scale, b.rw * cloud.scale, b.rh * cloud.scale, 8);
        ctx.fill();
      });
      ctx.restore();
    }

    // Canvas scaling to match container bounds sharp
    function resize() {
      const rect = canvas.getBoundingClientRect();
      canvas.width = rect.width * dpr;
      canvas.height = rect.height * dpr;
      ctx.resetTransform();
      ctx.scale(dpr, dpr);
      gridScale = Math.min(rect.width / 12.5, 38);
    }
    window.addEventListener('resize', resize);
    resize();

    // --- ENGINE TICK ---
    let lastTime = performance.now();
    function loop(now) {
      let dt = (now - lastTime) / 16.666;
      if (dt > 6) dt = 6; // Max clamp to prevent large physics skips on tab inactive
      lastTime = now;

      update(dt);
      render();
      requestAnimationFrame(loop);
    }
    requestAnimationFrame(loop);

    // Initial skins preview draw
    setTimeout(() => {
      renderSkinsPreview();
    }, 100);
  </script>
</body>
</html>
