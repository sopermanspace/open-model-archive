const COLS=13;
let W,H,DPR,TILE_W,TILE_H,centerX,centerY;
let cameraY=0, shake=0, shakeX=0, shakeY=0, time=0, lastTime=0;
let state='title';
let rows=[], particles=[], clouds=[];
let score=0, best=0, coins=0, startRow=2, maxRow=2;
let player, freezeTime=0;

function resize(){
  DPR=Math.min(window.devicePixelRatio||1,2);
  W=window.innerWidth*DPR; H=window.innerHeight*DPR;
  c.width=W; c.height=H;
  TILE_W=W*0.055; TILE_H=TILE_W*0.55;
  centerX=W/2; centerY=H*0.82;

function rand(a,b){ return a+Math.random()*(b-a); }
function randInt(a,b){ return Math.floor(rand(a,b+1)); }
function lerp(a,b,t){ return a+(b-a)*t; }
function shadeColor(col,p){
  const num=parseInt(col.replace('#',''),16);
  const amt=Math.round(2.55*p);
  let R=(num>>16)+amt,G=((num>>8)&0x00FF)+amt,B=(num&0x0000FF)+amt;
  R=R<0?0:R>255?255:R; G=G<0?0:G>255?255:G; B=B<0?0:B>255?255:B;
  return '#'+((1<<24)+(R<<16)+(G<<8)+B).toString(16).slice(1);
}

function iso(x,y,z=0){ return { x:(x-y)*TILE_W/2 + centerX + shakeX, y:(x+y)*TILE_H/2 + centerY + cameraY + shakeY - z*TILE_H*0.95 }; }
function drawBlock(x,y,z,w,l,h,topC){
  const s1=shadeColor(topC,-22), s2=shadeColor(topC,-40);
  const p000=iso(x,y,z), p100=iso(x+w,y,z), p010=iso(x,y+l,z), p110=iso(x+w,y+l,z);
  const p001=iso(x,y,z+h), p101=iso(x+w,y,z+h), p011=iso(x,y+l,z+h), p111=iso(x+w,y+l,z+h);
  ctx.fillStyle=s1; ctx.beginPath(); ctx.moveTo(p101.x,p101.y); ctx.lineTo(p100.x,p100.y); ctx.lineTo(p110.x,p110.y); ctx.lineTo(p111.x,p111.y); ctx.closePath(); ctx.fill();
  ctx.fillStyle=s2; ctx.beginPath(); ctx.moveTo(p011.x,p011.y); ctx.lineTo(p010.x,p010.y); ctx.lineTo(p110.x,p110.y); ctx.lineTo(p111.x,p111.y); ctx.closePath(); ctx.fill();
  ctx.fillStyle=topC; ctx.beginPath(); ctx.moveTo(p001.x,p001.y); ctx.lineTo(p101.x,p101.y); ctx.lineTo(p111.x,p111.y); ctx.lineTo(p011.x,p011.y); ctx.closePath(); ctx.fill();
}
function drawLine(a,b, color, width){
  ctx.strokeStyle=color; ctx.lineWidth=width; ctx.lineCap='round';
  ctx.beginPath(); ctx.moveTo(a.x,a.y); ctx.lineTo(b.x,b.y); ctx.stroke();
}

function randomCarColor(){ return ['#ef5350','#42a5f5','#ffca28','#66bb6a','#ab47bc','#26c6da','#ffa726'][randInt(0,6)]; }
function randomFlowerColor(){ return ['#e91e63','#ffeb3b','#9c27b0','#00bcd4'][randInt(0,3)]; }

  const g=ctx.createLinearGradient(0,0,0,H);
  g.addColorStop(0,'#0d1b3e'); g.addColorStop(0.45,'#6a4c93'); g.addColorStop(0.85,'#ff8a65'); g.addColorStop(1,'#ffcc80');
  ctx.fillStyle=g; ctx.fillRect(0,0,W,H);
}
function initClouds(){
  clouds=[];
  for(let i=0;i<8;i++) clouds.push({x:Math.random()*(W+400)-200, y:Math.random()*H*0.55, scale:0.4+Math.random()*1.3, speed:rand(4,18)});
}
function updateClouds(dt){
  clouds.forEach(cl=>{ cl.x-=cl.speed*dt*18; if(cl.x<-250) cl.x=W+250; });
}

    ctx.fillStyle='rgba(255,255,255,0.12)';
    ctx.beginPath();
    ctx.ellipse(cl.x,cl.y,70*cl.scale,28*cl.scale,0,0,Math.PI*2);
    ctx.ellipse(cl.x+45*cl.scale,cl.y+12*cl.scale,55*cl.scale,24*cl.scale,0,0,Math.PI*2);
    ctx.ellipse(cl.x-45*cl.scale,cl.y+6*cl.scale,50*cl.scale,22*cl.scale,0,0,Math.PI*2);
    ctx.fill();
  });
}

function chooseType(prev){
  const r=Math.random();
  if(prev==='water') return r<0.5?'grass':'road';
  if(prev==='rail') return r<0.55?'grass':'road';
  if(r<0.42) return 'grass';
  if(r<0.68) return 'road';
  if(r<0.87) return 'water';
  return 'rail';
}
function populateRoad(lane){
  const n=randInt(1,Math.min(4,2+Math.floor(score/120)));
  for(let i=0;i<n;i++){
    const isTruck=Math.random()<0.25;
    const length=isTruck?1.8:0.95;
    const dir=Math.random()<0.5?1:-1;
    const speed=isTruck?rand(1.6,2.6):rand(2.2,3.6);
    lane.vehicles.push({x:rand(-3,COLS+2), length, width:0.75, dir, speed, height:isTruck?0.7:0.5, color:randomCarColor()});
  }
}
function populateWater(lane){
  const n=randInt(2,Math.min(5,3+Math.floor(score/180)));
  for(let i=0;i<n;i++){
    const len=randInt(2,3);
    const dir=Math.random()<0.5?1:-1;
    const speed=rand(0.8,1.9);
    lane.logs.push({x:rand(-4,COLS+4), length:len, dir, speed});
  }
}
function populateGrass(lane){
  for(let col=0;col<COLS;col++) if(Math.random()<0.08) lane.flowers.push({col, color:randomFlowerColor()});
  if(Math.random()<0.35){
    const n=randInt(1,3);
    for(let i=0;i<n;i++){
      const col=randInt(0,COLS-1);
      if(lane.row===startRow && col===Math.floor(COLS/2)) continue;
      if(!lane.coins.includes(col)) lane.coins.push(col);
    }
  }
}
function populateLane(lane){
  if(lane.type==='road') populateRoad(lane);
  else if(lane.type==='water') populateWater(lane);
  else if(lane.type==='grass') populateGrass(lane);
  else if(lane.type==='rail') lane.railState={phase:'idle', timer:rand(2,4), trainX:-10, trainDir:1};
}
function addLane(r){
  const prev=rows.length?rows[rows.length-1].type:'grass';
  const type=r<=startRow?'grass':chooseType(prev);
  const lane={type,row:r,vehicles:[],logs:[],coins:[],flowers:[],railState:null,trainX:-10,trainDir:1};
  populateLane(lane);
  rows.push(lane);
}

function getLane(r){ for(let i=0;i<rows.length;i++) if(rows[i].row===r) return rows[i]; return null; }
function vehicleOverlaps(px,lane){
  const pMin=px+0.18, pMax=px+0.82;
  for(const v of lane.vehicles) if(pMax>v.x+0.05 && pMin<v.x+v.length-0.05) return true;
  return false;
}
function logUnder(px,lane){
  const pMin=px+0.18, pMax=px+0.82;
  for(const log of lane.logs) if(pMax>log.x && pMin<log.x+log.length) return log;
  return null;
}

function drawCoin(col,row){
  const cx=col+0.5, cy=row+0.5, z=0.28+Math.sin(time*7)*0.05;
  const p=iso(cx,cy,z);
  ctx.save(); ctx.translate(p.x,p.y); ctx.rotate(time*4);
  ctx.fillStyle='#ffd54f';
  ctx.beginPath(); ctx.ellipse(0,0,TILE_W*0.18,TILE_W*0.11,0,0,Math.PI*2); ctx.fill();
  ctx.fillStyle='#fff176';
  ctx.beginPath(); ctx.ellipse(-TILE_W*0.05,-TILE_W*0.03,TILE_W*0.06,TILE_W*0.04,0,0,Math.PI*2); ctx.fill();
  ctx.restore();
}
function drawVehicle(v,row){
  const y=row+0.12;
  drawBlock(v.x, y, 0, v.length, v.width, v.height, v.color);
  const headZ=v.height;
  const headColor='#fff9c4';
  if(v.dir>0) drawBlock(v.x+v.length-0.18, y+0.15, headZ, 0.14, v.width-0.3, 0.12, headColor);
  else drawBlock(v.x+0.04, y+0.15, headZ, 0.14, v.width-0.3, 0.12, headColor);
}
function drawLog(log,row){
  drawBlock(log.x, row+0.1, -0.9, log.length, 0.8, 0.28, '#8d6e63');
}
function drawTrain(lane){
  if(!lane.railState || lane.railState.phase!=='train') return;
  const len=COLS+10;
  drawBlock(lane.trainX, lane.row+0.05, 0, len, 0.9, 0.75, '#263238');
  const topZ=0.75;
  for(let i=2;i<len-1;i+=1.6) drawBlock(lane.trainX+i, lane.row+0.22, topZ, 0.6, 0.5, 0.12, '#ffeb3b');
}
function drawLane(lane){
  for(let col=0;col<COLS;col++){
    if(lane.type==='grass') drawBlock(col,lane.row,0,1,1,0.15,'#66bb6a');
    else if(lane.type==='road') drawBlock(col,lane.row,0,1,1,0.08,'#455a64');
    else if(lane.type==='water') drawBlock(col,lane.row,-2.4,1,1,1.8,'#29b6f6');
    else if(lane.type==='rail') drawBlock(col,lane.row,0,1,1,0.08,'#795548');
  }
  if(lane.type==='road'){
    ctx.globalAlpha=0.55;
    for(let col=0;col<COLS;col++) drawLine(iso(col+0.5,lane.row+0.4,0.12), iso(col+0.5,lane.row+0.6,0.12), '#fff', 2*DPR);
    ctx.globalAlpha=1;
  } else if(lane.type==='rail'){
    for(let col=0;col<COLS;col++){
      drawLine(iso(col,lane.row+0.25,0.12), iso(col+1,lane.row+0.25,0.12), '#90a4ae', 2*DPR);
      drawLine(iso(col,lane.row+0.75,0.12), iso(col+1,lane.row+0.75,0.12), '#90a4ae', 2*DPR);
      drawLine(iso(col+0.5,lane.row+0.25,0.12), iso(col+0.5,lane.row+0.75,0.12), '#5d4037', 2*DPR);
    }
    if(lane.railState && lane.railState.phase==='warning'){
      const flash=Math.sin(time*18)>0;
      drawBlock(-0.6,lane.row,0.32,COLS+1.2,0.16,0.12, flash?'#e53935':'#fff');
      drawBlock(-0.6,lane.row,0,0.3,0.3,0.55, flash?'#ff5252':'#fff');
      drawBlock(COLS+0.3,lane.row,0,0.3,0.3,0.55, flash?'#ff5252':'#fff');
    }
  } else if(lane.type==='grass'){
    lane.flowers.forEach(f=> drawBlock(f.col+0.35,lane.row+0.35,0.15,0.3,0.3,0.18,f.color));
  }
  if(lane.type==='road') lane.vehicles.forEach(v=>drawVehicle(v,lane.row));
  else if(lane.type==='water') lane.logs.forEach(log=>drawLog(log,lane.row));
  else if(lane.type==='rail') drawTrain(lane);
  lane.coins.forEach(col=>drawCoin(col,lane.row));
}
function drawShadow(cx,cy,z,scale=1){
  const p=iso(cx,cy,0);
  ctx.fillStyle='rgba(0,0,0,0.22)';
  ctx.beginPath(); ctx.ellipse(p.x,p.y,TILE_W*0.28*(1-z*0.18)*scale,TILE_H*0.28*(1-z*0.18)*scale,0,0,Math.PI*2); ctx.fill();
}
function playerScale(){
  if(player.dead) return {sx:1.2, sy:0.25};
  if(!player.hopping) return {sx:1, sy:1};
  const t=player.hopT;
  if(t<0.18){ const k=t/0.18; return {sx:1+0.26*(1-k), sy:1-0.3*(1-k)}; }
  if(t>0.82){ const k=(t-0.82)/0.18; return {sx:1+0.26*k, sy:1-0.3*k}; }
  const k=Math.sin((t-0.18)/0.64*Math.PI);
  return {sx:1-0.14*k, sy:1+0.22*k};
}
function drawPlayer(){
  const cx=player.renderX+0.5, cy=player.renderY+0.5, z=player.z;
  const sc=playerScale();
  drawShadow(cx,cy,z, sc.sy<0.5?0.7:1);
  const w=0.6*sc.sx, l=0.6*sc.sx, h=0.65*sc.sy;
  const bx=cx-w/2, by=cy-l/2;
  drawBlock(bx,by,z,w,l,h,'#ffffff');
  const ez=z+h, ew=0.18, el=0.18, eh=0.34*sc.sy;
  drawBlock(cx-0.22,cy-0.1,ez,ew,el,eh,'#ffffff');
  drawBlock(cx+0.04,cy-0.1,ez,ew,el,eh,'#ffffff');
  const eyeZ=z+h+0.015;
  ctx.fillStyle='#263238';
  const left=iso(cx-0.14,cy-0.02,eyeZ), right=iso(cx+0.14,cy-0.02,eyeZ);
  const sz=TILE_W*0.11;
  ctx.fillRect(left.x-sz/2,left.y-sz/2,sz,sz);
  ctx.fillRect(right.x-sz/2,right.y-sz/2,sz,sz);
  const nose=iso(cx,cy+0.08,eyeZ);
  ctx.fillStyle='#ff8a80';
  ctx.fillRect(nose.x-sz/3,nose.y-sz/3,sz*0.7,sz*0.7);
}

function spawnSparkles(x,y){
  for(let i=0;i<12;i++){
    const a=Math.random()*Math.PI*2, sp=rand(0.5,2.2);
    particles.push({x,y,z:0.5, vx:Math.cos(a)*sp, vy:Math.sin(a)*sp, vz:rand(1,3), life:0.55, max:0.55, color:Math.random()<0.5?'#ffd54f':'#fff', size:rand(2,4.5)*DPR});
  }
}
function spawnDeathParticles(x,y){
  const cols=['#ff5252','#ffffff','#ffca28','#ff8a80','#b0bec5'];
  for(let i=0;i<38;i++){
    const a=Math.random()*Math.PI*2, sp=rand(1.5,5);
    particles.push({x,y,z:0.5, vx:Math.cos(a)*sp, vy:Math.sin(a)*sp, vz:rand(2,5.5), life:0.85, max:0.85, color:cols[randInt(0,cols.length-1)], size:rand(3,7)*DPR});
  }
}
function updateParticles(dt){
  for(let i=particles.length-1;i>=0;i--){
    const p=particles[i];
    p.x+=p.vx*dt; p.y+=p.vy*dt; p.z+=p.vz*dt;
    p.vz-=8*dt;
    p.life-=dt;
    if(p.life<=0) particles.splice(i,1);
  }
}
function drawParticles(){
  particles.forEach(p=>{
    const pos=iso(p.x,p.y,Math.max(0,p.z));
    ctx.globalAlpha=Math.max(0,p.life/p.maxLife);
    ctx.fillStyle=p.color;
    ctx.fillRect(pos.x-p.size/2,pos.y-p.size/2,p.size,p.size);
  });
  ctx.globalAlpha=1;
}

function startHop(dx,dy){
  if(state!=='playing' || player.hopping || player.dead) return;
  const ty=player.y+dy; if(ty<0) return;
  let tx;
  if(dy!==0) tx=Math.round(player.x);
  else tx=Math.round(player.x)+dx;
  if(tx<0 || tx>=COLS) return;
  player.hopping=true; player.hopT=0; player.hopDir={x:dx,y:dy};
  player.hopStartX=player.x; player.hopStartY=player.y;
  player.hopTargetX=tx; player.hopTargetY=ty;
}
function updatePlayerHop(dt){
  if(!player.hopping){ player.renderX=player.x; player.renderY=player.y; player.z=0; return; }
  player.hopT+=dt/0.18;
  if(player.hopT>=1){
    player.hopT=1;
    player.x=player.hopTargetX; player.y=player.hopTargetY;
    player.hopping=false; player.renderX=player.x; player.renderY=player.y; player.z=0;
    handleLand();
    return;
  }
  const t=player.hopT;
  player.renderX=lerp(player.hopStartX,player.hopTargetX,t);
  player.renderY=lerp(player.hopStartY,player.hopTargetY,t);
  player.z=Math.sin(t*Math.PI)*0.55;
}
function handleLand(){
  if(player.dead || state!=='playing') return;
  const lane=getLane(player.y); if(!lane) return;
  if(lane.type==='road' && vehicleOverlaps(player.x,lane)){ die(); return; }
  if(lane.type==='water' && !logUnder(player.x,lane)){ die(); return; }
  if(lane.type==='rail' && lane.railState.phase==='train'){ die(); return; }
  if(player.y>maxRow){ maxRow=player.y; score=maxRow-startRow; if(score>best){ best=score; saveBest(); } }
  const ci=lane.coins.indexOf(Math.round(player.x));
  if(ci>=0){ lane.coins.splice(ci,1); coins++; score+=5; spawnSparkles(player.x+0.5,player.y+0.5); }
  updateHUD();
}
function die(){
  if(player.dead || state==='dying' || state==='gameover') return;
  player.dead=true; state='dying'; freezeTime=0.42; shake=16;
  spawnDeathParticles(player.renderX+0.5,player.renderY+0.5);
}

function updateLanes(dt){
  const mult=1+Math.min(score,600)/180;
  rows.forEach(lane=>{
    if(lane.type==='road'){
      lane.vehicles.forEach(v=>{
        v.x+=v.speed*v.dir*dt*mult;
        if(v.dir>0 && v.x>COLS+2) v.x=-v.length-2;
        if(v.dir<0 && v.x<-v.length-2) v.x=COLS+2;
      });
    } else if(lane.type==='water'){
      lane.logs.forEach(log=>{
        log.x+=log.speed*log.dir*dt*mult;
        if(log.dir>0 && log.x>COLS+3) log.x=-log.length-3;
        if(log.dir<0 && log.x<-log.length-3) log.x=COLS+3;
      });
      if(player.y===lane.row && !player.hopping && !player.dead){
        const log=logUnder(player.x,lane);
        if(log) player.x+=log.speed*log.dir*dt*mult;
        else die();
      }
    } else if(lane.type==='rail'){
      const st=lane.railState; st.timer-=dt;
      if(st.timer<=0){
        if(st.phase==='idle'){ st.phase='warning'; st.timer=1.2; }
        else if(st.phase==='warning'){ st.phase='train'; st.timer=1.35; st.trainDir=Math.random()<0.5?1:-1; st.trainX=st.trainDir>0?-10:COLS+10; }
        else { st.phase='idle'; st.timer=rand(2,4); }
      }
      if(st.phase==='train'){
        st.trainX+=st.trainDir*14*dt;
        if(player.y===lane.row && !player.dead) die();
      }
    }
  });
}
function generateRows(){
  while(rows.length && rows[0].row<player.y-30) rows.shift();
  while(rows[rows.length-1].row<player.y+35) addLane(rows[rows.length-1].row+1);
}

function resetGame(){
  rows=[]; particles=[]; score=0; coins=0; startRow=2; maxRow=startRow;
  const mid=Math.floor(COLS/2);
  player={x:mid,y:startRow,z:0,renderX:mid,renderY:startRow,hopping:false,hopT:0,hopDir:{x:0,y:0},hopStartX:mid,hopStartY:startRow,hopTargetX:mid,hopTargetY:startRow,dead:false};
  for(let r=0;r<=startRow+35;r++){
    addLane(r);
    if(r<=startRow){ rows[r].type='grass'; rows[r].vehicles=[]; rows[r].logs=[]; rows[r].railState=null; rows[r].coins=[]; rows[r].flowers=[]; populateGrass(rows[r]); }
  }
  cameraY=0; shake=0; freezeTime=0;
  updateHUD();
}
function initTitle(){

  for(let r=0;r<10;r++){
    const lane={type:'grass',row:r,vehicles:[],logs:[],coins:[],flowers:[],railState:null,trainX:-10,trainDir:1};
    rows.push(lane); populateGrass(lane);
  }
  const mid=Math.floor(COLS/2);

}

function renderWorld(){
  ctx.clearRect(0,0,W,H);

  for(let i=rows.length-1;i>=0;i--){
    const lane=rows[i];

function loop(now){
  const dt=Math.min((now-lastTime)/1000,0.05); lastTime=now; time+=dt;
  resize();
  shakeX=(Math.random()-0.5)*shake; shakeY=(Math.random()-0.5)*shake;
  if(state==='title'){
    player.z=Math.sin(time*2.5)*0.18; player.renderX=player.x; player.renderY=player.y;
    updateClouds(dt);
  } else if(state==='playing' || state==='dying'){
    if(state==='playing'){
      updateLanes(dt);
      if(!player.hopping && !player.dead){
        const lane=getLane(player.y);
        if(lane){
          if(lane.type==='road' && vehicleOverlaps(player.x,lane)) die();
          else if(lane.type==='water' && !logUnder(player.x,lane)) die();
          else if(lane.type==='rail' && lane.railState.phase==='train') die();
        }
      }
      updatePlayerHop(dt);
      generateRows();
      const pBase=iso(player.renderX+0.5,player.renderY+0.5,0);
      const baseNoCam=pBase.y-cameraY;
      const targetCam=H*0.75-baseNoCam;
      cameraY=lerp(cameraY,targetCam,0.12);
    }
    updateParticles(dt); updateClouds(dt);
    if(state==='dying'){
      freezeTime-=dt; shake=Math.max(0,shake-dt*35);
      if(freezeTime<=0){ state='gameover'; showGameover(); }
    }
  } else if(state==='gameover'){
    updateParticles(dt); updateClouds(dt);
    shake=Math.max(0,shake-dt*8);
  }
  renderWorld();
  requestAnimationFrame(loop);
}

  }
  touchStart=null;
},{passive:false});
