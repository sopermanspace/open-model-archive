    form.addEventListener('submit', function(e) {

      // Basic accessibility and standard validation
      if (!emailInput.validity.valid) {
        if (emailInput.validity.valueMissing) {
          errorMsg.textContent = "Please enter an email address so we can register you!";
        } else if (emailInput.validity.typeMismatch) {
          errorMsg.textContent = "That doesn't look like a valid space coordinate (email)!";
        }
        errorMsg.style.display = "block";
        emailInput.focus();
        return;
      }
      
      errorMsg.style.display = "none";
      const userEmail = emailInput.value;
      
      // Escape HTML to prevent injection issues
      const escapedEmail = userEmail.replace(/[&<>'"]/g, 
        tag => ({
          '&': '&amp;',
          '<': '&lt;',
          '>': '&gt;',
          "'": '&#39;',
          '"': '&quot;'
        }[tag] || tag)
      );

      // Render success card layout
      container.innerHTML = `
        <div class="success-card">
          <div class="success-icon">✨</div>
          <h3>Welcome onboard, Cadet!</h3>
          <p>We've registered <strong>${escapedEmail}</strong> to the crew manifest.</p>
